67 void Vertexer::initialise(const float zCut, const float phiCut, const float pairCut, const float clusterCut,
68  const int clusterContributorsCut)
69 {
70  for (int iLayer{ 0 }; iLayer < Constants::ITS::LayersNumberVertexer; ++iLayer) {
71  std::sort(mClusters[iLayer].begin(), mClusters[iLayer].end(), [](Cluster& cluster1, Cluster& cluster2) {
72  return cluster1.indexTableBinIndex < cluster2.indexTableBinIndex;
73  });
74  }
75  for (int iLayer{ 0 }; iLayer < Constants::ITS::LayersNumberVertexer; ++iLayer) {
76  const int clustersNum = static_cast<int>(mClusters[iLayer].size());
77  int previousBinIndex{ 0 };
78  mIndexTables[iLayer][0] = 0;
79  for (int iCluster{ 0 }; iCluster < clustersNum; ++iCluster) {
80  const int currentBinIndex{ mClusters[iLayer][iCluster].indexTableBinIndex };
81  if (currentBinIndex > previousBinIndex) {
82  for (int iBin{ previousBinIndex + 1 }; iBin <= currentBinIndex; ++iBin) {
83  mIndexTables[iLayer][iBin] = iCluster;
84  }
85  previousBinIndex = currentBinIndex;
86  }
87  }
88  for (int iBin{ previousBinIndex + 1 }; iBin <= ZBins * PhiBins; iBin++) {
89  mIndexTables[iLayer][iBin] = clustersNum;
90  }
91  }
92 
93  mZCut = zCut;
94  mPhiCut = phiCut;
95  mPairCut = pairCut;
96  mClusterCut = clusterCut;
97  mClusterContributorsCut = clusterContributorsCut;
98  mPhiSpan = static_cast<int>(std::ceil(PhiBins * mPhiCut / TwoPi));
99  mZSpan = static_cast<int>(std::ceil(mZCut * Constants::IndexTable::InverseZBinSize()[0]));
100  mVertexerInitialised = true;
101 }

351 void Vertexer::findVertices()
352 {
353  if (mTrackletsFound) {
354  const int numTracklets{ static_cast<int>(mTracklets.size()) };
355  mUsedTracklets.resize(numTracklets, false);
356 
357  for (int tracklet1{ 0 }; tracklet1 < numTracklets; ++tracklet1) {
358  if (mUsedTracklets[tracklet1])
359  continue;
360  for (int tracklet2{ tracklet1 + 1 }; tracklet2 < numTracklets; ++tracklet2) {
361  if (mUsedTracklets[tracklet2])
362  continue;
363  if (Line::getDCA(mTracklets[tracklet1], mTracklets[tracklet2]) <= mPairCut) {
364  mTrackletClusters.emplace_back(tracklet1, mTracklets[tracklet1], tracklet2, mTracklets[tracklet2]);
365  std::array<float, 3> tmpVertex{ mTrackletClusters.back().getVertex() };
366  if (tmpVertex[0] * tmpVertex[0] + tmpVertex[1] * tmpVertex[1] > 4.f) {
367  mTrackletClusters.pop_back();
368  break;
369  }
370  mUsedTracklets[tracklet1] = true;
371  mUsedTracklets[tracklet2] = true;
372  for (int tracklet3{ 0 }; tracklet3 < numTracklets; ++tracklet3) {
373  if (mUsedTracklets[tracklet3])
374  continue;
375  if (Line::getDistanceFromPoint(mTracklets[tracklet3], tmpVertex) < mPairCut) {
376  mTrackletClusters.back().add(tracklet3, mTracklets[tracklet3]);
377  mUsedTracklets[tracklet3] = true;
378  tmpVertex = mTrackletClusters.back().getVertex();
379  }
380  }
381  break;
382  }
383  }
384  }
385  std::sort(mTrackletClusters.begin(), mTrackletClusters.end(),
386  [](ClusterLines& cluster1, ClusterLines& cluster2) { return cluster1.getSize() > cluster2.getSize(); });
387  for (int iCluster1{ 0 }; iCluster1 < mTrackletClusters.size(); ++iCluster1) {
388  std::array<float, 3> vertex1{ mTrackletClusters[iCluster1].getVertex() };
389  std::array<float, 3> vertex2{};
390  for (int iCluster2{ iCluster1 + 1 }; iCluster2 < mTrackletClusters.size(); ++iCluster2) {
391  vertex2 = mTrackletClusters[iCluster2].getVertex();
392  if (std::abs(vertex1[2] - vertex2[2]) < mClusterCut) {
393  float distance{ (vertex1[0] - vertex2[0]) * (vertex1[0] - vertex2[0]) +
394  (vertex1[1] - vertex2[1]) * (vertex1[1] - vertex2[1]) +
395  (vertex1[2] - vertex2[2]) * (vertex1[2] - vertex2[2]) };
396  if (distance <= mPairCut * mPairCut) {
397  for (auto label : mTrackletClusters[iCluster2].getLabels()) {
398  mTrackletClusters[iCluster1].add(label, mTracklets[label]);
399  vertex1 = mTrackletClusters[iCluster1].getVertex();
400  }
401  mTrackletClusters.erase(mTrackletClusters.begin() + iCluster2);
402  --iCluster2;
403  }
404  }
405  }
406  for (int iCluster{ 0 }; iCluster < mTrackletClusters.size(); ++iCluster) {
407  if (mTrackletClusters[iCluster].getSize() < mClusterContributorsCut && mTrackletClusters.size() > 1) {
408  mTrackletClusters.erase(mTrackletClusters.begin() + iCluster);
409  continue;
410  }
411 
412  mVertices.emplace_back(mTrackletClusters[iCluster].getVertex());
413  }
414  }
415  }
