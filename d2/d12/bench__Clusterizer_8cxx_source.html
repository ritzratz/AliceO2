23 std::vector<o2::mid::ColumnData> generateTestData(int deId, const o2::mid::Mapping& midMapping)
24 {

26 
27  std::uniform_int_distribution<int> distColumn(firstColumnInDE, 6);
28  std::uniform_int_distribution<int> distStrip(0, 15);
29  std::uniform_int_distribution<int> distNfired(0, 4);
30 
31  std::random_device rd;
32  std::mt19937 mt(rd());
33 
34  std::vector<o2::mid::ColumnData> patterns;
35  int firstColumn = distColumn(mt);
36  int lastColumn = firstColumn + 1;
37  if (lastColumn > 6) {
38  lastColumn = 6;
39  }
40 
41  for (int icol = firstColumn; icol <= lastColumn; ++icol) {
42  o2::mid::ColumnData column;
43  column.deId = (uint8_t)deId;
44  column.columnId = (uint8_t)icol;
45  for (int cathode = 0; cathode < 2; ++cathode) {
46  int firstLine = 0;
47  int lastLine = 0;
48  if (cathode == 0) {
49  firstLine = midMapping.getFirstBoardBP(icol, deId);
50  lastLine = midMapping.getLastBoardBP(icol, deId);
51  }
52  for (int iline = firstLine; iline <= lastLine; ++iline) {
53  uint16_t pattern = 0;
54  int nStrips = distNfired(mt);
55  for (int istrip = 0; istrip < nStrips; ++istrip) {
56  if (midMapping.stripByLocation(istrip, cathode, iline, column.columnId, column.deId).isValid()) {
57  pattern |= (1 << istrip);
58  }
59  }
60  if (cathode == 0) {
61  column.patterns.setBendPattern(pattern, iline);
62  } else {
63  column.patterns.setNonBendPattern(pattern);
64  }
65  }
66  }
67  patterns.emplace_back(column);
68  }
69  return patterns;
70 }
