47 bool Propagator::PropagateToXBxByBz(o2::track::TrackParCov& track, float xToGo, float mass, float maxSnp, float maxStep,
48  int matCorr, int signCorr)
49 {
50  //----------------------------------------------------------------
51  //
52  // Propagates the track to the plane X=xk (cm)
53  // taking into account all the three components of the magnetic field
54  // and correcting for the crossed material.
55  //
56  // mass - mass used in propagation - used for energy loss correction (if <0 then q=2)
57  // maxStep - maximal step for propagation
58  //
59  //----------------------------------------------------------------
60  const float Epsilon = 0.00001;
61  auto dx = xToGo - track.getX();
62  int dir = dx > 0.f ? 1 : -1;
63  if (!signCorr) {
64  signCorr = -dir; // sign of eloss correction is not imposed
65  }
66 
67  std::array<float, 3> b;
68  while (std::abs(dx) > Epsilon) {
69  auto step = std::min(std::abs(dx), maxStep);
70  if (dir < 0) {
71  step = -step;
72  }
73  auto x = track.getX() + step;
74  auto xyz0 = track.getXYZGlo();
75  mField->Field(xyz0, b.data());
76 
77  if (!track.propagateTo(x, b))
78  return false;
79  if (maxSnp > 0 && std::abs(track.getSnp()) >= maxSnp)
80  return false;
81 
82  if (matCorr) {
83  auto xyz1 = track.getXYZGlo();
84  auto mb = GeometryManager::MeanMaterialBudget(xyz0, xyz1);
85  if (signCorr < 0) {
86  mb.length = -mb.length;
87  }
88  //
89  if (!track.correctForMaterial(mb.meanX2X0, mb.meanRho * mb.length, mass))
90  return false;
91  }
92  dx = xToGo - track.getX();
93  }
94  return true;
95 }
