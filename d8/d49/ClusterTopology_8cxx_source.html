42  unsigned int ClusterTopology::hashFunction(const void* key, int len)
43  {
44  //
45  // Developed from https://github.com/rurban/smhasher , function MurMur2
46  //
47  // 'm' and 'r' are mixing constants generated offline.
48  const unsigned int m = 0x5bd1e995;
49  const int r = 24;
50  // Initialize the hash
51  unsigned int h = len ^ 0xdeadbeef;
52  // Mix 4 bytes at a time into the hash
53  const unsigned char* data = (const unsigned char*)key;
54  // int recIndex=0;
55  while (len >= 4) {
56  unsigned int k = *(unsigned int*)data;
57  k *= m;
58  k ^= k >> r;
59  k *= m;
60  h *= m;
61  h ^= k;
62  data += 4;
63  len -= 4;
64  }
65  // Handle the last few bytes of the input array
66  switch (len) {
67  case 3:
68  h ^= data[2] << 16;
69  case 2:
70  h ^= data[1] << 8;
71  case 1:
72  h ^= data[0];
73  h *= m;
74  };
75  // Do a few final mixes of the hash to ensure the last few
76  // bytes are well-incorporated.
77  h ^= h >> 13;
78  h *= m;
79  h ^= h >> 15;
80  return h;
81  }

83  unsigned long ClusterTopology::getCompleteHash(int nRow, int nCol,
84  const unsigned char patt[Cluster::kMaxPatternBytes], int nBytesUsed)
85  {
86  unsigned long extended_pattern[Cluster::kMaxPatternBytes + 2] = { 0 };
87  extended_pattern[0] = (unsigned char)nRow;
88  extended_pattern[1] = (unsigned char)nCol;
89  memcpy(&extended_pattern[2], patt, nBytesUsed);
90 
91  unsigned long partialHash = (unsigned long)hashFunction(extended_pattern, nBytesUsed);
92  // The first four bytes are directly taken from partialHash
93  unsigned long completeHash = partialHash << 32;
94  // The last four bytes of the hash are the first 32 pixels of the topology.
95  // The bits reserved for the pattern that are not used are set to 0.
96  if (nBytesUsed == 1) {
97  completeHash += ((((unsigned long)extended_pattern[2]) << 24));
98  } else if (nBytesUsed == 2) {
99  completeHash += ((((unsigned long)extended_pattern[2]) << 24) + (((unsigned long)extended_pattern[3]) << 16));
100  } else if (nBytesUsed == 3) {
101  completeHash += ((((unsigned long)extended_pattern[2]) << 24) + (((unsigned long)extended_pattern[3]) << 16) +
102  (((unsigned long)extended_pattern[4]) << 8));
103  } else if (nBytesUsed >= 4) {
104  completeHash += ((((unsigned long)extended_pattern[2]) << 24) + (((unsigned long)extended_pattern[3]) << 16) +
105  (((unsigned long)extended_pattern[4]) << 8) + ((unsigned long)extended_pattern[5]));
106  } else {
107  std::cout << "ERROR: no fired pixels\n";
108  exit(1);
109  }
110  return completeHash;
111  }
