74 bool PreClusterFinderDevice::processData(FairMQMessagePtr& msg, int /*index*/)
75 {
78 
79  static const AliHLTComponentDataType digitBlockDataType = AliHLTComponentDataTypeInitializer("DIGITS ", "MUON");
80 
81  // prepare to receive new data
82  mFormatHandler.clear();
83  mPreClusterFinder.reset();
84  bool validBlockFound(false);
85 
86  if (msg->GetSize() > 0) {
87 
88  // convert the message in ALICE HLT format
89  if (mFormatHandler.addMessage(reinterpret_cast<AliHLTUInt8_t*>(msg->GetData()), msg->GetSize()) > 0) {
90 
91  vector<BlockDescriptor>& inputBlocks = mFormatHandler.getBlockDescriptors();
92  for (const auto& block : inputBlocks) {
93 
94  // only digit blocks
95  if (!MatchExactly(block.fDataType, digitBlockDataType)) {
96  LOG(INFO) << "not a MUON digit block: " << block.fDataType.fID;
97  continue;
98  }
99 
100  // load the digits to get the fired pads
101  auto digitBlock(reinterpret_cast<const DigitBlock*>(block.fPtr));
102  auto digits(reinterpret_cast<const DigitStruct*>(digitBlock + 1));
103  mPreClusterFinder.loadDigits(digits, digitBlock->header.fNrecords);
104 
105  validBlockFound = true;
106  }
107 
108  } else {
109  LOG(WARN) << "no valid data blocks in message";
110  }
111 
112  } else {
113  LOG(WARN) << "ignoring message with payload of size 0";
114  }
115 
116  if (validBlockFound) {
117 
118  // preclusterize
119  int nPreClusters = mPreClusterFinder.run();
120 
121  if (nPreClusters > 0) {
122 
123  // number of DEs with preclusters and total number of pads used
124  int nUsedDigits(0);
125  int nDEWithPreClusters = mPreClusterFinder.getNDEWithPreClusters(nUsedDigits);
126 
127  // create message of the exactly needed buffer size
128  auto size = nDEWithPreClusters * sizeof(AliHLTComponentBlockData) +
129  PreClusterBlock::sizeOfPreClusterBlocks(nDEWithPreClusters, nPreClusters, nUsedDigits);
130  FairMQMessagePtr msgOut(NewMessage(size));
131 
132  // store preclusters
133  try {
134  storePreClusters(reinterpret_cast<uint8_t*>(msgOut->GetData()), msgOut->GetSize());
135  } catch (exception const& e) {
136  LOG(ERROR) << "PreClusterFinder::processData() failed storing preclusters: " << e.what();
137  return false;
138  }
139 
140  // Send out the output message
141  if (Send(msgOut, "data-out") < 0) {
142  LOG(ERROR) << "problem sending message";
143  return false;
144  }
145  }
146  }
147 
148  // return true if want to be called again (otherwise go to IDLE state)
149  return true;
150 }

169 void PreClusterFinderDevice::storePreClusters(uint8_t* buffer, uint32_t size)
170 {
172 
173  const PreClusterFinder::PreCluster* cluster(nullptr);
174  const DigitStruct* digit(nullptr);
175  uint32_t totalBytesUsed(0);
176 
177  // loop over DEs
178  for (int iDE = 0, nDEs = mPreClusterFinder.getNDEs(); iDE < nDEs; ++iDE) {
179 
180  if (!mPreClusterFinder.hasPreClusters(iDE)) {
181  continue;
182  }
183 
184  // create the preclusters data block for this DE
185  AliHLTComponentBlockData* blockData(nullptr);
186  if (size - totalBytesUsed >= sizeof(AliHLTComponentBlockData)) {
187  blockData = reinterpret_cast<AliHLTComponentBlockData*>(buffer + totalBytesUsed);
188  fillBlockData(*blockData);
189  totalBytesUsed += sizeof(AliHLTComponentBlockData);
190  } else {
191  LOG(ERROR) << "The buffer is too small to store the data block.";
192  throw overflow_error("The buffer is too small to store the data block.");
193  }
194 
195  if (mPreClusterBlock.reset(buffer + totalBytesUsed, size - totalBytesUsed, true) < 0) {
196  throw runtime_error("Cannot reset the cluster block.");
197  }
198 
199  // loop over planes
200  for (int iPlane = 0; iPlane < 2; ++iPlane) {
201 
202  // loop over preclusters
203  for (int iCluster = 0, nClusters = mPreClusterFinder.getNPreClusters(iDE, iPlane); iCluster < nClusters;
204  ++iCluster) {
205 
206  cluster = mPreClusterFinder.getPreCluster(iDE, iPlane, iCluster);
207  if (!cluster->storeMe) {
208  continue;
209  }
210 
211  // add the precluster with its first digit
212  digit = mPreClusterFinder.getDigit(iDE, cluster->firstPad);
213  if (mPreClusterBlock.startPreCluster(*digit) < 0) {
214  throw runtime_error("Cannot store a new precluster.");
215  }
216 
217  // loop over other pads and add corresponding digits
218  for (uint16_t iOrderedPad = cluster->firstPad + 1; iOrderedPad <= cluster->lastPad; ++iOrderedPad) {
219  digit = mPreClusterFinder.getDigit(iDE, iOrderedPad);
220  if (mPreClusterBlock.addDigit(*digit) < 0) {
221  throw runtime_error("Cannot store a new digit.");
222  }
223  }
224  }
225  }
226 
227  // complete the data block information
228  uint32_t bytesUsed(mPreClusterBlock.getCurrentSize());
229  blockData->fSize = bytesUsed;
230  blockData->fSpecification = mPreClusterFinder.getDEId(iDE);
231  totalBytesUsed += bytesUsed;
232  }
233 }
