50 float TrackingUtils::computeCurvature(float x1, float y1, float x2, float y2, float x3, float y3)
51 {
52  const float d = (x2 - x1) * (y3 - y2) - (x3 - x2) * (y2 - y1);
53  const float a =
54  0.5f * ((y3 - y2) * (y2 * y2 - y1 * y1 + x2 * x2 - x1 * x1) - (y2 - y1) * (y3 * y3 - y2 * y2 + x3 * x3 - x2 * x2));
55  const float b =
56  0.5f * ((x2 - x1) * (y3 * y3 - y2 * y2 + x3 * x3 - x2 * x2) - (x3 - x2) * (y2 * y2 - y1 * y1 + x2 * x2 - x1 * x1));
57 
58  return -1.f * d / std::sqrt((d * x1 - a) * (d * x1 - a) + (d * y1 - b) * (d * y1 - b));
59 }

61 float TrackingUtils::computeCurvatureCentreX(float x1, float y1, float x2, float y2, float x3, float y3)
62 {
63  const float k1 = (y2 - y1) / (x2 - x1), k2 = (y3 - y2) / (x3 - x2);
64  return 0.5f * (k1 * k2 * (y1 - y3) + k2 * (x1 + x2) - k1 * (x2 + x3)) / (k2 - k1);
65 }
