198 bool MatchTPCITS::validateTPCMatch(int mtID)
199 {
200  auto& tpcMatch = mMatchesTPC[mtID];
201  auto& rcTPC = mMatchRecordsTPC[tpcMatch.first]; // best TPC->ITS match
202  if (rcTPC.nextRecID == Validated)
203  return false; // RS do we need this
204  // check if it is consistent with corresponding ITS->TPC match
205  auto& itsMatch = mMatchesITS[rcTPC.matchID]; // matchCand of partner ITS track
206  auto& rcITS = mMatchRecordsITS[itsMatch.first]; // best ITS->TPC match
207  if (rcTPC.nextRecID == Validated)
208  return false; // RS do we need this ?
209  if (rcITS.matchID == mtID) { // is best matching TPC track for this ITS track actually mtID?
210 
211  // unlink winner TPC track from all ITS candidates except winning one
212  int nextTPC = rcTPC.nextRecID;
213  while (nextTPC > MinusOne) {
214  auto& rcTPCrem = mMatchRecordsTPC[nextTPC];
215  removeTPCfromITS(mtID, rcTPCrem.matchID); // remove references on mtID from ITS match=rcTPCrem.matchID
216  nextTPC = rcTPCrem.nextRecID;
217  }
218  rcTPC.nextRecID = Validated;
219  int itsWinID = rcTPC.matchID;
220 
221  // unlink winner ITS match from all TPC matches using it
222  int nextITS = rcITS.nextRecID;
223  while (nextITS > MinusOne) {
224  auto& rcITSrem = mMatchRecordsITS[nextITS];
225  removeITSfromTPC(itsWinID, rcITSrem.matchID); // remove references on itsWinID from TPC match=rcITSrem.matchID
226  nextITS = rcITSrem.nextRecID;
227  }
228  rcITS.nextRecID = Validated;
229  return true;
230  }
231  return false;
232 }

313 bool MatchTPCITS::prepareTPCTracks()
314 {
316  mMatchRecordsTPC.clear();
317 
318  if (!loadTPCTracksNextChunk()) {
319  return false;
320  }
321 
322  int ntr = mTPCTracksArrayInp->size();
323 
324  mMatchesTPC.reserve(mMatchesTPC.size() + ntr);
325  // number of records might be actually more than N tracks!
326  mMatchRecordsTPC.reserve(mMatchRecordsTPC.size() + mMaxMatchCandidates * ntr);
327 
328  // copy the track params, propagate to reference X and build sector tables
329  mTPCWork.clear();
330  mTPCWork.reserve(ntr);
331  if (mMCTruthON) {
332  mTPCLblWork.clear();
333  mTPCLblWork.reserve(ntr);
334  }
335  for (int sec = o2::constants::math::NSectors; sec--;) {
336  mTPCSectIndexCache[sec].clear();
337  mTPCSectIndexCache[sec].reserve(100 + 1.2 * ntr / o2::constants::math::NSectors);
338  mTPCTimeBinStart[sec].clear();
339  }
340 
341  for (int it = 0; it < ntr; it++) {
342 
343  o2::TPC::TrackTPC& trcOrig = (*mTPCTracksArrayInp)[it];
344 
345  // make sure the track was propagated to inner TPC radius at the ref. radius
346  if (trcOrig.getX() > mXTPCInnerRef + 0.1)
347  continue; // failed propagation to inner TPC radius, cannot be matched
348 
349  // create working copy of track param
350  mTPCWork.emplace_back(static_cast<o2::track::TrackParCov&>(trcOrig), mCurrTPCTracksTreeEntry, it);
351  auto& trc = mTPCWork.back();
352  // propagate to matching Xref
353  if (!propagateToRefX(trc)) {
354  mTPCWork.pop_back(); // discard track whose propagation to mXRef failed
355  continue;
356  }
357  if (mMCTruthON) {
358  mTPCLblWork.emplace_back(mTPCTrkLabels->getLabels(it)[0]);
359  }
360 
361  float time0 = trcOrig.getTime0() - mNTPCBinsFullDrift;
362  trc.timeBins.set(time0 - trcOrig.getDeltaTBwd() - mTPCTimeEdgeTSafeMargin,
363  time0 + trcOrig.getDeltaTFwd() + mTPCTimeEdgeTSafeMargin);
364  // assign min max possible Z for this track which still respects the clusters A/C side
365  if (trcOrig.hasASideClustersOnly()) {
366  trc.zMin = trc.getZ() - trcOrig.getDeltaTBwd() * mTPCBin2Z;
367  trc.zMax = trc.getZ() + trcOrig.getDeltaTFwd() * mTPCBin2Z;
368  } else if (trcOrig.hasCSideClustersOnly()) {
369  trc.zMin = trc.getZ() - trcOrig.getDeltaTFwd() * mTPCBin2Z;
370  trc.zMax = trc.getZ() + trcOrig.getDeltaTBwd() * mTPCBin2Z;
371  }
372  // TODO : special treatment of tracks crossing the CE
373 
374  // cache work track index
375  mTPCSectIndexCache[o2::utils::Angle2Sector(trc.getAlpha())].push_back(mTPCWork.size() - 1);
376  }
377 
378  // sort tracks in each sector according to their timeMax
379  for (int sec = o2::constants::math::NSectors; sec--;) {
380  auto& indexCache = mTPCSectIndexCache[sec];
381  LOG(INFO) << "Sorting sector" << sec << " | " << indexCache.size() << " TPC tracks" << FairLogger::endl;
382  if (!indexCache.size())
383  continue;
384  std::sort(indexCache.begin(), indexCache.end(), [this](int a, int b) {
385  auto& trcA = mTPCWork[a];
386  auto& trcB = mTPCWork[b];
387  return (trcA.timeBins.tmax - trcB.timeBins.tmax) < 0.;
388  });
389 
390  // build array of 1st entries with tmax corresponding to each ITS RO cycle
391  float tmax = mTPCWork[indexCache.back()].timeBins.tmax;
392  int nbins = 1 + tpcTimeBin2ITSROFrame(tmax);
393  auto& tbinStart = mTPCTimeBinStart[sec];
394  tbinStart.resize(nbins > 1 ? nbins : 1, -1);
395  tbinStart[0] = 0;
396  for (int itr = 0; itr < (int)indexCache.size(); itr++) {
397  auto& trc = mTPCWork[indexCache[itr]];
398  int bTrc = tpcTimeBin2ITSROFrame(trc.timeBins.tmax);
399  if (bTrc < 0) {
400  continue;
401  }
402  if (tbinStart[bTrc] == -1) {
403  tbinStart[bTrc] = itr;
404  }
405  }
406  for (int i = 1; i < nbins; i++) {
407  if (tbinStart[i] == -1) { // fill gaps with preceding indices
408  tbinStart[i] = tbinStart[i - 1];
409  }
410  }
411  } // loop over tracks of single sector
412 
413  return true;
414 }

417 bool MatchTPCITS::prepareITSTracks()
418 {
419  // load next chunk of ITS data and prepare for matching
420  mMatchesITS.clear();
421  mITSWork.clear();
422  // number of records might be actually more than N tracks!
423  mMatchRecordsITS.clear(); // RS TODO reserve(mMatchRecordsITS.size() + mMaxMatchCandidates*ntr);
424  if (mMCTruthON) {
425  mITSLblWork.clear();
426  }
427  for (int sec = o2::constants::math::NSectors; sec--;) {
428  mITSSectIndexCache[sec].clear();
429  }
430 
431  while (loadITSTracksNextChunk()) {
432  int ntr = mITSTracksArrayInp->size();
433  for (int it = 0; it < ntr; it++) {
434  auto& trcOrig = (*mITSTracksArrayInp)[it];
435 
436  if (trcOrig.getParamOut().getX() < 1.) {
437  continue; // backward refit failed
438  }
439  // working copy of outer track param
440  mITSWork.emplace_back(static_cast<o2::track::TrackParCov&>(trcOrig.getParamOut()), mCurrITSTracksTreeEntry, it);
441  auto& trc = mITSWork.back();
442 
443  // TODO: why I did this?
444  if (!trc.rotate(o2::utils::Angle2Alpha(trc.getPhiPos()))) {
445  mITSWork.pop_back(); // discard failed track
446  continue;
447  }
448  // make sure the track is at the ref. radius
449  if (!propagateToRefX(trc)) {
450  mITSWork.pop_back(); // discard failed track
451  continue; // add to cache only those ITS tracks which reached ref.X and have reasonable snp
452  }
453  if (mMCTruthON) {
454  mITSLblWork.emplace_back(mITSTrkLabels->getLabels(it)[0]);
455  }
456 
457  float tmn = itsROFrame2TPCTimeBin(trcOrig.getROFrame());
458  trc.timeBins.set(tmn, tmn + mITSROFrame2TPCBin);
459  trc.roFrame = trcOrig.getROFrame();
460 
461  // cache work track index
462  int sector = o2::utils::Angle2Sector(trc.getAlpha());
463  mITSSectIndexCache[sector].push_back(mITSWork.size() - 1);
464 
465  // If the ITS track is very close to the sector edge, it may match also to a TPC track in the neighb. sector.
466  // For a track with Yr and Phir at Xr the distance^2 between the poisition of this track in the neighb. sector
467  // when propagated to Xr (in this neighbouring sector) and the edge will be (neglecting the curvature)
468  // [(Xr*tg(10)-Yr)/(tgPhir+tg70)]^2 / cos(70)^2 // for the next sector
469  // [(Xr*tg(10)+Yr)/(tgPhir-tg70)]^2 / cos(70)^2 // for the prev sector
470  // Distances to the sector edges in neighbourings sectors (at Xref in theit proper frames)
471  float tgp = trc.getSnp();
472  tgp /= std::sqrt((1.f - tgp) * (1.f + tgp)); // tan of track direction XY
473 
474  // sector up
475  float dy2Up = (mYMaxAtXRef - trc.getY()) / (tgp + Tan70);
476  if ((dy2Up * dy2Up * Cos70I2) < mSectEdgeMargin2) { // need to check this track for matching in sector up
477  addTrackCloneForNeighbourSector(trc, sector < (o2::constants::math::NSectors - 1) ? sector + 1 : 0);
478  }
479  // sector down
480  float dy2Dn = (mYMaxAtXRef + trc.getY()) / (tgp - Tan70);
481  if ((dy2Dn * dy2Dn * Cos70I2) < mSectEdgeMargin2) { // need to check this track for matching in sector down
482  addTrackCloneForNeighbourSector(trc, sector > 1 ? sector - 1 : o2::constants::math::NSectors - 1);
483  }
484  }
485  }
486  // sort tracks in each sector according to their time, then tgl
487  for (int sec = o2::constants::math::NSectors; sec--;) {
488  auto& indexCache = mITSSectIndexCache[sec];
489  LOG(INFO) << "Sorting sector" << sec << " | " << indexCache.size() << " ITS tracks" << FairLogger::endl;
490  if (!indexCache.size()) {
491  continue;
492  }
493  std::sort(indexCache.begin(), indexCache.end(), [this](int a, int b) {
494  auto& trackA = mITSWork[a];
495  auto& trackB = mITSWork[b];
496  if (trackA.roFrame < trackB.roFrame) { // ITS tracks have the same time coverage
497  return true;
498  } else if (trackA.roFrame > trackB.roFrame) {
499  return false;
500  }
501  return trackA.getTgl() < trackB.getTgl();
502  });
503 
504  // build array of 1st entries with of each ITS RO cycle
505  int nbins = 1 + mITSWork[indexCache.back()].roFrame;
506  auto& tbinStart = mITSTimeBinStart[sec];
507  tbinStart.resize(nbins > 1 ? nbins : 1, -1);
508  tbinStart[0] = 0;
509  for (int itr = 0; itr < (int)indexCache.size(); itr++) {
510  auto& trc = mITSWork[indexCache[itr]];
511  if (tbinStart[trc.roFrame] == -1) {
512  tbinStart[trc.roFrame] = itr;
513  }
514  }
515  for (int i = 1; i < nbins; i++) {
516  if (tbinStart[i] == -1) { // fill gaps with preceding indices
517  tbinStart[i] = tbinStart[i - 1];
518  }
519  }
520  } // loop over tracks of single sector
521  mMatchesITS.reserve(mITSWork.size());
522  mMatchRecordsITS.reserve(mITSWork.size() * mMaxMatchCandidates);
523 
524  return true;
525 }

570 //_____________________________________________________
571 void MatchTPCITS::doMatching(int sec)
572 {
574  auto& cacheITS = mITSSectIndexCache[sec]; // array of cached ITS track indices for this sector
575  auto& cacheTPC = mTPCSectIndexCache[sec]; // array of cached ITS track indices for this sector
576  auto& tbinStartTPC = mTPCTimeBinStart[sec]; // array of 1st TPC track with timeMax in ITS ROFrame
577  auto& tbinStartITS = mITSTimeBinStart[sec];
578  int nTracksTPC = cacheTPC.size(), nTracksITS = cacheITS.size();
579  if (!nTracksTPC || !nTracksITS) {
580  LOG(INFO) << "Matchng sector " << sec << " : N tracks TPC:" << nTracksTPC << " ITS:" << nTracksITS << " in sector "
581  << sec << FairLogger::endl;
582  return;
583  }
584 
586  float maxTDriftSafe = (mNTPCBinsFullDrift + mITSTPCTimeBinSafeMargin + mTPCTimeEdgeTSafeMargin);
587 
588  // get min ROFrame (in TPC time-bins) of ITS tracks currently in cache
589  auto minROFITS = mITSWork[cacheITS.front()].roFrame;
590 
591  if (minROFITS >= int(tbinStartTPC.size())) {
592  LOG(INFO) << "ITS min ROFrame " << minROFITS << " exceeds all cached TPC track ROF eqiuvalent "
593  << cacheTPC.size() - 1 << FairLogger::endl;
594  return;
595  }
596 
597  int nCheckTPCControl = 0, nCheckITSControl = 0, nMatchesControl = 0; // temporary
598 
599  int idxMinTPC = tbinStartTPC[minROFITS]; // index of 1st cached TPC track within cached ITS ROFrames
600  for (int itpc = idxMinTPC; itpc < nTracksTPC; itpc++) {
601  auto& trefTPC = mTPCWork[cacheTPC[itpc]];
602  // estimate ITS 1st ROframe bin this track may match to: TPC track are sorted according to their
603  // timeMax, hence the timeMax - MaxmNTPCBinsFullDrift are non-decreasing
604  int itsROBin = tpcTimeBin2ITSROFrame(trefTPC.timeBins.tmax - maxTDriftSafe);
605  if (itsROBin >= int(tbinStartITS.size())) { // time of TPC track exceeds the max time of ITS in the cache
606  break;
607  }
608  int iits0 = tbinStartITS[itsROBin];
609  nCheckTPCControl++;
610  for (auto iits = iits0; iits < nTracksITS; iits++) {
611  auto& trefITS = mITSWork[cacheITS[iits]];
612  // compare if the ITS and TPC tracks may overlap in time
613  if (trefTPC.timeBins.tmax < trefITS.timeBins.tmin) {
614  // since TPC tracks are sorted in timeMax and ITS tracks are sorted in timeMin
615  // all following ITS tracks also will not match
616  break;
617  }
618  if (trefTPC.timeBins.tmin > trefITS.timeBins.tmax) { // its bracket is fully before TPC bracket
619  continue;
620  }
621  nCheckITSControl++;
622  float chi2 = -1;
623  int rejFlag = compareITSTPCTracks(trefITS, trefTPC, chi2);
624 
625 #ifdef _ALLOW_DEBUG_TREES_
626  if (mDBGOut && ((rejFlag == Accept && isDebugFlag(MatchTreeAccOnly)) || isDebugFlag(MatchTreeAll))) {
627  fillITSTPCmatchTree(cacheITS[iits], cacheTPC[itpc], rejFlag, chi2);
628  }
629 #endif
630 
631  if (rejFlag == RejectOnTgl) {
632  // ITS tracks in each ROFrame are ordered in Tgl, hence if this check failed on Tgl check
633  // (i.e. tgl_its>tgl_tpc+tolerance), tnem all other ITS tracks in this ROFrame will also have tgl too large.
634  // Jump on the 1st ITS track of the next ROFrame
635  int rof = trefITS.roFrame;
636  bool stop = false;
637  do {
638  if (++rof >= int(tbinStartITS.size())) {
639  stop = true;
640  break; // no more ITS ROFrames in cache
641  }
642  iits = tbinStartITS[rof] - 1; // next track to be checked -1
643  } while (iits <= tbinStartITS[trefITS.roFrame]); // skip empty bins
644  if (stop) {
645  break;
646  }
647  continue;
648  }
649  if (rejFlag != Accept) {
650  continue;
651  }
652  mTimerReg.Start(false);
653  registerMatchRecordTPC(trefITS, trefTPC, chi2); // register matching candidate
654  mTimerReg.Stop();
655  nMatchesControl++;
656  }
657  }
658 
659  LOG(INFO) << "Match sector " << sec << " N tracks TPC:" << nTracksTPC << " ITS:" << nTracksITS
660  << " N TPC tracks checked: " << nCheckTPCControl << " (starting from " << idxMinTPC
661  << "), checks: " << nCheckITSControl << ", matches:" << nMatchesControl << FairLogger::endl;
662 }

686 //______________________________________________
687 bool MatchTPCITS::registerMatchRecordTPC(TrackLocITS& tITS, TrackLocTPC& tTPC, float chi2)
688 {
691 
692  auto& mtcTPC = getTPCMatchEntry(tTPC); // get matchCand structure of this TPC track, create if none
693  int nextID = mtcTPC.first; // get 1st matchRecord this matchCand refers to
694  if (nextID < 0) { // no matches yet, just add new record
695  registerMatchRecordITS(tITS, tTPC.matchID, chi2); // register matchCand entry in the ITS records
696  mtcTPC.first = mMatchRecordsTPC.size(); // new record will be added in the end
697  mMatchRecordsTPC.emplace_back(tITS.matchID, chi2); // create new record with empty reference on next match
698  return true;
699  }
700 
701  int count = 0, topID = MinusOne;
702  do {
703  auto& nextMatchRec = mMatchRecordsTPC[nextID];
704  count++;
705  if (chi2 < nextMatchRec.chi2) { // need to insert new record before nextMatchRec?
706  if (count < mMaxMatchCandidates) {
707  break; // will insert in front of nextID
708  } else { // max number of candidates reached, will overwrite the last one
709  nextMatchRec.chi2 = chi2;
710  suppressMatchRecordITS(nextMatchRec.matchID, tTPC.matchID); // flag as disabled the overriden ITS match
711  registerMatchRecordITS(tITS, tTPC.matchID, chi2); // register matchCand entry in the ITS records
712  nextMatchRec.matchID = tITS.matchID; // reuse the record of suppressed ITS match to store better one
713  return true;
714  }
715  }
716  topID = nextID; // check next match record
717  nextID = nextMatchRec.nextRecID;
718  } while (nextID > MinusOne);
719 
720  // if count == mMaxMatchCandidates, the max number of candidates was already reached, and the
721  // new candidated was either discarded (if its chi2 is worst one) or has overwritten worst
722  // existing candidate. Otherwise, we need to add new entry
723  if (count < mMaxMatchCandidates) {
724  if (topID < 0) { // the new match is top candidate
725  topID = mtcTPC.first = mMatchRecordsTPC.size(); // register new record as top one
726  } else { // there are better candidates
727  topID = mMatchRecordsTPC[topID].nextRecID = mMatchRecordsTPC.size(); // register to his parent
728  }
729  // nextID==-1 will mean that the while loop run over all candidates->the new one is the worst (goes to the end)
730  registerMatchRecordITS(tITS, tTPC.matchID, chi2); // register matchCand entry in the ITS records
731  mMatchRecordsTPC.emplace_back(tITS.matchID, chi2, nextID); // create new record with empty reference on next match
732  // make sure that after addition the number of candidates don't exceed allowed number
733  count++;
734  while (nextID > MinusOne) {
735  if (count > mMaxMatchCandidates) {
736  suppressMatchRecordITS(mMatchRecordsTPC[nextID].matchID, tTPC.matchID);
737  // exclude nextID record, w/o changing topID (which becomes the last record)
738  nextID = mMatchRecordsTPC[topID].nextRecID = mMatchRecordsTPC[nextID].nextRecID;
739  continue;
740  }
741  count++;
742  topID = nextID;
743  nextID = mMatchRecordsTPC[nextID].nextRecID;
744  }
745  return true;
746  } else {
747  return false; // unless nextID was assigned OverrideExisting, new candidate was discarded
748  }
749 }

751 //______________________________________________
752 void MatchTPCITS::registerMatchRecordITS(TrackLocITS& tITS, int matchTPCID, float chi2)
753 {
755  auto& itsMatch = getITSMatchEntry(tITS); // if needed, create new entry
756  int nextRecord = itsMatch.first; // entry of 1st match record in mMatchRecordsITS
757  int idnew = mMatchRecordsITS.size();
758  mMatchRecordsITS.emplace_back(matchTPCID, chi2); // associate index of matchCand with this record
759  if (nextRecord < 0) { // this is the 1st match for this TPC track
760  itsMatch.first = idnew;
761  return;
762  }
763  // there are other matches for this ITS track, insert the new record preserving chi2 order
764  // navigate till last record or the one with worse chi2
765  int topID = MinusOne;
766  auto& newRecord = mMatchRecordsITS.back();
767  do {
768  auto& recITS = mMatchRecordsITS[nextRecord];
769  if (chi2 < recITS.chi2) { // insert before this one
770  newRecord.nextRecID = nextRecord; // new one will refer to old one it overtook
771  if (topID < 0) {
772  itsMatch.first = idnew; // the new one is the best match, the matchCand will refer to it
773  } else {
774  mMatchRecordsITS[topID].nextRecID = idnew; // new record will follow existing better one
775  }
776  return;
777  }
778  topID = nextRecord;
779  nextRecord = mMatchRecordsITS[nextRecord].nextRecID;
780  } while (nextRecord > MinusOne);
781 
782  // if we reached here, the new record should be added in the end
783  mMatchRecordsITS[topID].nextRecID = idnew; // register new link
784 }

786 //______________________________________________
787 int MatchTPCITS::compareITSTPCTracks(const TrackLocITS& tITS, const TrackLocTPC& tTPC, float& chi2) const
788 {
790  auto& trackTPC = tTPC;
791  auto& trackITS = tITS;
792  chi2 = -1.f;
793  int rejFlag = Accept;
794  float diff; // make rough check differences and their nsigmas
795 
796  // start with check on Tgl, since rjection on it will allow to profit from sorting
797  diff = trackITS.getParam(o2::track::kTgl) - trackTPC.getParam(o2::track::kTgl);
798  if ((rejFlag = roughCheckDif(diff, mCrudeAbsDiffCut[o2::track::kTgl], RejectOnTgl))) {
799  return rejFlag;
800  }
801  diff *= diff / (trackITS.getDiagError2(o2::track::kTgl) + trackTPC.getDiagError2(o2::track::kTgl));
802  if ((rejFlag = roughCheckDif(diff, mCrudeNSigma2Cut[o2::track::kTgl], RejectOnTgl + NSigmaShift))) {
803  return rejFlag;
804  }
805 
806  diff = trackITS.getParam(o2::track::kY) - trackTPC.getParam(o2::track::kY);
807  if ((rejFlag = roughCheckDif(diff, mCrudeAbsDiffCut[o2::track::kY], RejectOnY))) {
808  return rejFlag;
809  }
810  diff *= diff / (trackITS.getDiagError2(o2::track::kY) + trackTPC.getDiagError2(o2::track::kY));
811  if ((rejFlag = roughCheckDif(diff, mCrudeNSigma2Cut[o2::track::kY], RejectOnY + NSigmaShift))) {
812  return rejFlag;
813  }
814 
815  if (mCompareTracksDZ) { // in continuous mode we usually don't use DZ
816  diff = trackITS.getParam(o2::track::kZ) - trackTPC.getParam(o2::track::kZ);
817  if ((rejFlag = roughCheckDif(diff, mCrudeAbsDiffCut[o2::track::kZ], RejectOnZ))) {
818  return rejFlag;
819  }
820  diff *= diff / (trackITS.getDiagError2(o2::track::kZ) + trackTPC.getDiagError2(o2::track::kZ));
821  if ((rejFlag = roughCheckDif(diff, mCrudeNSigma2Cut[o2::track::kZ], RejectOnZ + NSigmaShift))) {
822  return rejFlag;
823  }
824  } else { // in continuous mode we use special check on allowed Z range
825  if (trackITS.getParam(o2::track::kZ) - tTPC.zMax > mCrudeAbsDiffCut[o2::track::kZ])
826  return RejectOnZ;
827  if (trackITS.getParam(o2::track::kZ) - tTPC.zMin < -mCrudeAbsDiffCut[o2::track::kZ])
828  return -RejectOnZ;
829  }
830 
831  diff = trackITS.getParam(o2::track::kSnp) - trackTPC.getParam(o2::track::kSnp);
832  if ((rejFlag = roughCheckDif(diff, mCrudeAbsDiffCut[o2::track::kSnp], RejectOnSnp))) {
833  return rejFlag;
834  }
835  diff *= diff / (trackITS.getDiagError2(o2::track::kSnp) + trackTPC.getDiagError2(o2::track::kSnp));
836  if ((rejFlag = roughCheckDif(diff, mCrudeNSigma2Cut[o2::track::kSnp], RejectOnSnp + NSigmaShift))) {
837  return rejFlag;
838  }
839 
840  diff = trackITS.getParam(o2::track::kQ2Pt) - trackTPC.getParam(o2::track::kQ2Pt);
841  if ((rejFlag = roughCheckDif(diff, mCrudeAbsDiffCut[o2::track::kQ2Pt], RejectOnQ2Pt))) {
842  return rejFlag;
843  }
844  diff *= diff / (trackITS.getDiagError2(o2::track::kQ2Pt) + trackTPC.getDiagError2(o2::track::kQ2Pt));
845  if ((rejFlag = roughCheckDif(diff, mCrudeNSigma2Cut[o2::track::kQ2Pt], RejectOnQ2Pt + NSigmaShift))) {
846  return rejFlag;
847  }
848  // calculate mutual chi2 excluding Z in continuos mode
849  chi2 = getPredictedChi2NoZ(tITS, tTPC);
850  if (chi2 > mCutMatchingChi2)
851  return RejectOnChi2;
852 
853  return Accept;
854 }

922 float MatchTPCITS::getPredictedChi2NoZ(const o2::track::TrackParCov& tr1, const o2::track::TrackParCov& tr2) const

927  // if (std::abs(tr1.getAlpha() - tr2.getAlpha()) > FLT_EPSILON) {

929  // << tr1.getAlpha() << " : " << tr2.getAlpha() << FairLogger::endl;

932  // if (std::abs(tr1.getX() - tr2.getX()) > FLT_EPSILON) {

938  covMat(0, 0) = static_cast<double>(tr1.getSigmaY2()) + static_cast<double>(tr2.getSigmaY2());
939  covMat(1, 0) = static_cast<double>(tr1.getSigmaSnpY()) + static_cast<double>(tr2.getSigmaSnpY());

946  covMat(3, 2) = static_cast<double>(tr1.getSigma1PtTgl()) + static_cast<double>(tr2.getSigma1PtTgl());
947  covMat(3, 3) = static_cast<double>(tr1.getSigma1Pt2()) + static_cast<double>(tr2.getSigma1Pt2());
948  if (!covMat.Invert()) {
949  LOG(ERROR) << "Cov.matrix inversion failed: " << covMat << FairLogger::endl;
950  return 2. * o2::track::HugeF;
951  }
952  double chi2diag = 0., chi2ndiag = 0.,
953  diff[o2::track::kNParams - 1] = { tr1.getParam(o2::track::kY) - tr2.getParam(o2::track::kY),
954  tr1.getParam(o2::track::kSnp) - tr2.getParam(o2::track::kSnp),
955  tr1.getParam(o2::track::kTgl) - tr2.getParam(o2::track::kTgl),

957  for (int i = o2::track::kNParams - 1; i--;) {
958  chi2diag += diff[i] * diff[i] * covMat(i, i);
959  for (int j = i; j--;) {
960  chi2ndiag += diff[i] * diff[j] * covMat(i, j);
961  }
962  }
963  return chi2diag + 2. * chi2ndiag;
964 }

1095 bool MatchTPCITS::refitTrackITSTPC(const TrackLocITS& tITS)
1096 {
1098  if (tITS.matchID < 0 || isDisabledITS(mMatchesITS[tITS.matchID])) {
1099  return false; // no match
1100  }
1101  const auto& itsMatch = mMatchesITS[tITS.matchID];
1102  const auto& itsMatchRec = mMatchRecordsITS[itsMatch.first];
1103  const auto& tTPC = mTPCWork[mTPCMatch2Track[itsMatchRec.matchID]];
1104 
1105  loadITSClustersChunk(tITS.source.getEvent());
1106  loadITSTracksChunk(tITS.source.getEvent());
1107  loadTPCTracksChunk(tTPC.source.getEvent());
1108 
1109  mMatchedTracks.emplace_back(tTPC); // create a copy of TPC track at xRef
1110  auto& trfit = mMatchedTracks.back();
1111  // in continuos mode the Z of TPC track is meaningless, unless it is CE crossing
1112  // track (currently absent, TODO)
1113  if (!mCompareTracksDZ) {
1114  trfit.setZ(tITS.getZ()); // fix the seed Z
1115  }
1116  float deltaT = (trfit.getZ() - tTPC.getZ()) * mZ2TPCBin; // time correction in time-bins
1117 
1118  auto itsTrOrig = (*mITSTracksArrayInp)[tITS.source.getIndex()]; // currently we store clusterIDs in the track
1119  int nclRefit = 0, ncl = itsTrOrig.getNumberOfClusters();
1120  float chi2 = 0.f;
1121  auto geom = o2::ITS::GeometryTGeo::Instance();
1122  auto propagator = o2::Base::Propagator::Instance();
1123  for (int icl = 0; icl < ncl; icl++) {
1124  const auto& clus = (*mITSClustersArrayInp)[itsTrOrig.getClusterIndex(icl)];
1125  float alpha = geom->getSensorRefAlpha(clus.getSensorID()), x = clus.getX();
1126  if (!trfit.rotate(alpha) ||
1127  !propagator->PropagateToXBxByBz(trfit, x, o2::constants::physics::MassPionCharged, MaxSnp, 2., 1)) {
1128  break;
1129  }
1130  chi2 += trfit.getPredictedChi2(static_cast<const o2::BaseCluster<float>&>(clus));
1131  if (!trfit.update(static_cast<const o2::BaseCluster<float>&>(clus))) {
1132  break;
1133  }
1134  nclRefit++;
1135  }
1136  if (nclRefit != ncl) {
1137  printf("FAILED after ncl=%d\n", nclRefit);
1138  printf("its was: ");
1139  tITS.print();
1140  printf("tpc was: ");
1141  tTPC.print();
1142  mMatchedTracks.pop_back(); // destroy failed track
1143  return false;
1144  }
1145 
1147  auto tpcTrOrig = (*mTPCTracksArrayInp)[tTPC.source.getIndex()];
1148  float time = tpcTrOrig.getTime0() - mNTPCBinsFullDrift;
1149  if (tpcTrOrig.hasASideClustersOnly()) {
1150  time += deltaT;
1151  } else if (tpcTrOrig.hasCSideClustersOnly()) {
1152  time -= deltaT;
1153  } else {
1154  // TODO : special treatment of tracks crossing the CE
1155  }
1156  // convert time to microseconds
1157  time *= mTPCTBinMUS;
1158  // estimate the error on time
1159  float timeErr = std::sqrt(tITS.getSigmaZ2() + tTPC.getSigmaZ2()) * mTPCVDrift0Inv;
1160  trfit.setChi2Match(itsMatchRec.chi2);
1161  trfit.setChi2Refit(chi2);
1162  trfit.setTimeMUS(time, timeErr);
1163  trfit.setRefTPC(tTPC.source);
1164  trfit.setRefITS(tITS.source);
1165  return true;
1166 }
