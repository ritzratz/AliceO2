154 template <typename T, typename... Targs>
155 constexpr T String2(char c, Targs... Fargs)
156 {
157  // number of arguments has either to match the size of the type, or the last element is treated
158  // as '0' if missing
159  // TODO: this is a bit of a question, do we allow the number of characters to
160  // be smaller than the size of the type and just pad with 0 like in the case
161  // of a char array argument?
162  static_assert(sizeof...(Targs) == sizeof(T) + 1 ||
163  sizeof...(Targs) == sizeof(T),
164  "number of arguments does not match the uint type width"
165  );
166  return internal::String2__<T>(c, Fargs...);
167 }

173 template <typename T, std::size_t N, std::size_t pos = 0, bool suppressAssert = false>
174 constexpr T String2(const char (&str)[N])
175 {
176  static_assert(std::is_integral<T>::value, "Non integral types not compatible with String2<type>");
177  static_assert(N >= pos, "Position is larger than the length of string");
178  static_assert(suppressAssert || N - pos <= sizeof(T) + 1,
179  "String parameter is longer than the size of the data type"
180  );
181  return((T) str[0+pos] |
182  (str[0+pos] && sizeof(T) >= 2 ? ((T) str[1+pos] << (sizeof(T) >= 2 ? 8 : 0) |
183  (str[1+pos] && sizeof(T) >= 4 ? ((T) str[2+pos] << (sizeof(T) >= 4 ? 16 : 0) |
184  (str[2+pos] && sizeof(T) >= 4 ? ((T) str[3+pos] << (sizeof(T) >= 4 ? 24 : 0) |
185  (str[3+pos] && sizeof(T) >= 8 ? ((T) str[4+pos] << (sizeof(T) >= 8 ? 32 : 0) |
186  (str[4+pos] && sizeof(T) >= 8 ? ((T) str[5+pos] << (sizeof(T) >= 8 ? 40 : 0) |
187  (str[5+pos] && sizeof(T) >= 8 ? ((T) str[6+pos] << (sizeof(T) >= 8 ? 48 : 0) |
188  (str[6+pos] && sizeof(T) >= 8 ? ((T) str[7+pos] << (sizeof(T) >= 8 ? 56 : 0) )
189  : 0)) : 0)) : 0)) : 0)) : 0)) : 0)) : 0));
190 }

273  void runtimeInit(const char* string, short length = -1) {
274  char* target = str;
275  char* targetEnd = target;
276  if (length >= 0 && length < (int)N) targetEnd += length;
277  else targetEnd += N;
278  const char* source = string;
279  for ( ; source != nullptr && target < targetEnd && *source !=0; ++target, ++source) *target = *source;
280  targetEnd = str + N;
281  for ( ; target < targetEnd; ++target) *target = 0;
282  // require the string to be not longer than the descriptor size
283  assert(source != nullptr && (*source == 0 || (length >= 0 && length <= (int)N)));
284  }
