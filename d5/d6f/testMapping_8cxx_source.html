59  const std::vector<Mapping::MpStripIndex>& neighbours, const Mapping& mapping)
60 {
61  int found = 0;

63  Mapping::MpStripIndex refStrip = mapping.stripByPosition(xPos, yPos + delta, cathode, deId, false);
64  if (!refStrip.isValid()) {
65  return found;

67

72

76  if (!findInNeighbours(refStrip, neighbours, cathode)) {

85 bool areNeighboursOk(const Mapping::MpStripIndex& stripIndex, const std::vector<Mapping::MpStripIndex>& neighbours,
86  const Mapping& mapping, int cathode, int deId)
87 {
88  MpArea stripArea =
89  mapping.stripByLocation(stripIndex.strip, cathode, stripIndex.line, stripIndex.column, deId, false);
90 
91  std::stringstream ss;
92  ss << "Strip " << stripIndex.strip << " cathode " << cathode << " line " << stripIndex.line << " column "
93  << stripIndex.column;
94 
95  bool isOk = true;
96 
97  double xPos[3] = { stripArea.getXmin() - 0.1, stripArea.getCenterX(), stripArea.getXmax() + 0.1 };
98  double yPos[3] = { stripArea.getYmin() - 0.1, stripArea.getCenterY(), stripArea.getYmax() + 0.1 };
99 
100  int nMatched = 0;
101  bool checkUpDown = false;
102  for (int ix = 0; ix < 3; ++ix) {
103  for (int iy = 0; iy < 3; ++iy) {
104  if ((ix - 1) * (iy - 1) != 0 || ix == iy) {
105  continue;
106  }
107  if (cathode == 0) {
108  checkUpDown = (ix == 1) ? false : true;
109  } else if (iy != 1) {
110  continue;
111  }
112 
113  int nFound = findInNeighbours(xPos[ix], yPos[iy], cathode, deId, checkUpDown, neighbours, mapping);
114  if (nFound < 0) {
115  ss << " err: (" << ix - 1 << ", " << iy - 1 << ")";
116  isOk = false;
117  } else {
118  nMatched += nFound;
119  }
120  }
121  }
122 
123  if (nMatched != neighbours.size()) {
124  isOk = false;
125  ss << " matched " << nMatched << " != " << neighbours.size();
126  }
127 
128  if (!isOk) {
129  std::cerr << ss.str() << ". Neighbours: " << std::endl;
130  for (auto const& neigh : neighbours) {
131  std::cerr << " strip " << neigh.strip << " line " << neigh.line << " column " << neigh.column << std::endl;
132  }
133  }
134 
135  return isOk;
136 }
