256 std::vector<Point> checkGaps(const Segmentation& seg, double xstep = 1.0, double ystep = 1.0)
257 {
258  std::vector<Point> gaps;
259  auto bbox = o2::mch::mapping::getBBox(seg);
260  auto env = o2::mch::mapping::getEnvelop(seg);
261 
262  if (env.size() != 1) {
263  throw std::runtime_error("assumption env contour = one polygon is not verified");
264  }
265 
266  for (double x = bbox.xmin() - xstep; x <= bbox.xmax() + xstep; x += xstep) {
267  for (double y = bbox.ymin() - ystep; y <= bbox.ymax() + ystep; y += ystep) {
268  double distanceToEnveloppe = std::sqrt(o2::mch::contour::squaredDistancePointToPolygon({ x, y }, env[0]));
269  bool withinEnveloppe = env.contains(x, y) && (distanceToEnveloppe > 1E-5);
270  if (withinEnveloppe && !seg.isValid(seg.findPadByPosition(x, y))) {
271  gaps.push_back(std::make_pair(x, y));
272  }
273  }
274  }
275  return gaps;
276 }
