44 std::vector<Event> IOUtils::loadEventData(const std::string& fileName)
45 {
46  std::vector<Event> events{};
47  std::ifstream inputStream{};
48  std::string line{}, unusedVariable{};
49  int layerId{}, monteCarlo{};
50  int clusterId{ EventLabelsSeparator };
51  float xCoordinate{}, yCoordinate{}, zCoordinate{}, alphaAngle{};
52  float varZ{ -1.f }, varY{ -1.f };
53 
54  inputStream.open(fileName);
55 
57  dataformats::MCTruthContainer<MCCompLabel>* mcLabels = nullptr;
58  while (std::getline(inputStream, line)) {
59 
60  std::istringstream inputStringStream(line);
61  if (inputStringStream >> layerId >> xCoordinate >> yCoordinate >> zCoordinate) {
62 
63  if (layerId == PrimaryVertexLayerId) {
64 
65  if (clusterId != 0) {
66  events.emplace_back(events.size());
67  }
68 
69  events.back().addPrimaryVertex(xCoordinate, yCoordinate, zCoordinate);
70  clusterId = 0;
71 
72  } else {
73 
74  if (inputStringStream >> varY >> varZ >> unusedVariable >> alphaAngle >> monteCarlo) {
75  events.back().addClusterToLayer(layerId, xCoordinate, yCoordinate, zCoordinate,
76  events.back().getLayer(layerId).getClustersSize());
77  const float sinAlpha = std::sin(alphaAngle);
78  const float cosAlpha = std::cos(alphaAngle);
79  const float xTF = xCoordinate * cosAlpha - yCoordinate * sinAlpha;
80  const float yTF = xCoordinate * sinAlpha + yCoordinate * cosAlpha;
81  events.back().addTrackingFrameInfoToLayer(layerId, xTF, alphaAngle, std::array<float, 2>{ yTF, zCoordinate },
82  std::array<float, 3>{ varY, 0.f, varZ });
83  events.back().addClusterLabelToLayer(layerId, MCCompLabel(monteCarlo));
84 
85  ++clusterId;
86  }
87  }
88  }
89  }
90 
91  return events;
92 }

154 std::vector<std::unordered_map<int, Label>> IOUtils::loadLabels(const int eventsNum, const std::string& fileName)
155 {
156  std::vector<std::unordered_map<int, Label>> labelsMap{};
157  std::unordered_map<int, Label> currentEventLabelsMap{};
158  std::ifstream inputStream{};
159  std::string line{};
160  int monteCarloId{}, pdgCode{}, numberOfClusters{};
161  float transverseMomentum{}, phiCoordinate{}, pseudorapidity{};
162 
163  labelsMap.reserve(eventsNum);
164 
165  inputStream.open(fileName);
166  std::getline(inputStream, line);
167 
168  while (std::getline(inputStream, line)) {
169 
170  std::istringstream inputStringStream(line);
171 
172  if (inputStringStream >> monteCarloId) {
173 
174  if (monteCarloId == EventLabelsSeparator) {
175 
176  labelsMap.emplace_back(currentEventLabelsMap);
177  currentEventLabelsMap.clear();
178 
179  } else {
180 
181  if (inputStringStream >> transverseMomentum >> phiCoordinate >> pseudorapidity >> pdgCode >> numberOfClusters) {
182 
183  if (std::abs(pdgCode) == Constants::PDGCodes::PionCode && numberOfClusters == 7) {
184 
185  currentEventLabelsMap.emplace(std::piecewise_construct, std::forward_as_tuple(monteCarloId),
186  std::forward_as_tuple(monteCarloId, transverseMomentum, phiCoordinate,
187  pseudorapidity, pdgCode, numberOfClusters));
188  }
189  }
190  }
191  }
192  }
193 
194  labelsMap.emplace_back(currentEventLabelsMap);
195 
196  return labelsMap;
