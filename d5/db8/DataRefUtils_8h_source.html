29  // SFINAE makes this available only for the case we are using
30  // trivially copyable type, this is to distinguish it from the
31  // alternative below, which works for TObject (which are serialised).
32  template <typename T>
33  static typename std::enable_if<is_messageable<T>::value == true, gsl::span<T>>::type
34  as(DataRef const& ref)
35  {
36  using DataHeader = o2::header::DataHeader;
37  auto header = o2::header::get<const DataHeader>(ref.header);
38  if (header->payloadSerializationMethod != o2::header::gSerializationMethodNone) {
39  throw std::runtime_error("Attempt to extract a POD from a wrong message kind");
40  }
41  if ((header->payloadSize % sizeof(T)) != 0) {
42  throw std::runtime_error("Cannot extract POD from message as size do not match");
43  }
44  //FIXME: provide a const collection
45  return gsl::span<T>(reinterpret_cast<T *>(const_cast<char *>(ref.payload)), header->payloadSize/sizeof(T));
46  }
47 
48  // See above. SFINAE allows us to use this to extract a ROOT-serialized object
49  // with a somewhat uniform API.
50  // Classes using the ROOT ClassDef/ClassImp macros can be detected automatically
51  // and the serialization method can be deduced. If the class is also
52  // messageable, gSerializationMethodNone is used by default. This substitution
53  // does not apply for such types, the serialization method needs to be specified
54  // explicitely using type wrapper @a ROOTSerialized.
55  template <typename T>
56  static typename std::enable_if<has_root_dictionary<T>::value == true &&
57  is_messageable<T>::value == false,
58  std::unique_ptr<T>>::type
59  as(DataRef const& ref)
60  {
61  using DataHeader = o2::header::DataHeader;
62  auto header = o2::header::get<const DataHeader>(ref.header);
63  if (header->payloadSerializationMethod != o2::header::gSerializationMethodROOT) {
64  throw std::runtime_error("Attempt to extract a TMessage from non-ROOT serialised message");
65  }
66 
67  o2::framework::FairTMessage ftm(const_cast<char*>(ref.payload), header->payloadSize);
68  auto* storedClass = ftm.GetClass();
69  auto* requestedClass = TClass::GetClass(typeid(T));
70  // should always have the class description if has_root_dictionary is true
71  assert(requestedClass != nullptr);
72 
73  auto* object = ftm.ReadObjectAny(storedClass);
74  if (object == nullptr) {
75  std::ostringstream ss;
76  ss << "Failed to read object with name "
77  << (storedClass != nullptr ? storedClass->GetName() : "<unknown>")
78  << " from message using ROOT serialization.";
79  throw std::runtime_error(ss.str());
80  }
81 
82  std::unique_ptr<T> result;
83  if (std::is_base_of<TObject, T>::value) { // compile time switch
84  // if the type to be extracted inherits from TObject, the class descriptions
85  // do not need to match exactly, only the dynamic_cast has to work
86  // FIXME: could probably try to extend this to arbitrary types T being a base
87  // to the stored type, but this would require to cast the extracted object to
88  // the actual type. This requires this information to be available as a type
89  // in the ROOT dictionary, check if this is the case or if TClass::DynamicCast
90  // can be used for the test. Right now, this case was and is not covered and
91  // not yet checked in the unit test either.
92  auto* r = dynamic_cast<T*>(static_cast<TObject*>(object));
93  if (r) {
94  result.reset(r);
95  }
96  } else if (storedClass == requestedClass) {
97  result.reset(static_cast<T*>(object));
98  }
99  if (result == nullptr) {
100  // did not manage to cast the pointer to result
101  // delete object via the class info, at this point we know that the
102  // class info is there, the object could not have been read otherwise
103  auto* delfunc = storedClass->GetDelete();
104  assert(delfunc != nullptr);
105  if (delfunc) (*delfunc)(object);
106 
107  std::ostringstream ss;
108  ss << "Attempting to extract a "
109  << (requestedClass != nullptr ? requestedClass->GetName() : "<unknown>")
110  << " but a "
111  << (storedClass != nullptr ? storedClass->GetName() : "<unknown>")
112  << " is actually stored which cannot be casted to the requested one.";
113  throw std::runtime_error(ss.str());
114  }
115 
116  return std::move(result);
117  }
118 
119  // See above. SFINAE allows us to use this to extract a ROOT-serialized object
120  // with a somewhat uniform API. ROOT serialization method is enforced by using
121  // type wrapper @a ROOTSerialized
122  template <typename W>
123  static typename std::enable_if<is_specialization<W, ROOTSerialized>::value == true,
124  std::unique_ptr<typename W::wrapped_type>>::type
125  as(DataRef const& ref)
126  {
127  using T = typename W::wrapped_type;
128  using DataHeader = o2::header::DataHeader;
129  auto header = o2::header::get<const DataHeader>(ref.header);
130  if (header->payloadSerializationMethod != o2::header::gSerializationMethodROOT) {
131  throw std::runtime_error("Attempt to extract a TMessage from non-ROOT serialised message");
132  }
133  auto* cl = TClass::GetClass(typeid(T));
134  if (has_root_dictionary<T>::value == false && cl == nullptr) {
135  throw std::runtime_error("ROOT serialization not supported, dictionary not found for data type");
136  }
137 
138  o2::framework::FairTMessage ftm(const_cast<char*>(ref.payload), header->payloadSize);
139  std::unique_ptr<T> result(static_cast<T*>(ftm.ReadObjectAny(cl)));
140  if (result.get() == nullptr) {
141  std::ostringstream ss;
142  ss << "Unable to extract class ";
143  if (cl == nullptr) {
144  ss << "<name not available>";
145  } else {
146  ss << cl->GetName();
147  }
148  throw std::runtime_error(ss.str());
149  }
150  return std::move(result);
151  }

157  if (!header) {
