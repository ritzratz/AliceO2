49 void Mapping::setupSegmentation(int rpcType, int column, int nStripsNBP, int stripPitchNBP, int nBoardsBP,
50  int firstBoardId, bool isBelowBeamPipe)
51 {
53  MpColumn& columnStruct{ mDetectionElements[rpcType].columns[column] };
54  columnStruct.nStripsNBP = nStripsNBP;
55  columnStruct.stripPitchNBP = stripPitchNBP;
56  columnStruct.stripPitchBP = (nBoardsBP > 0) ? 4 / nBoardsBP : 0;
57  int nBoardsBPtot = nBoardsBP;
58  if (nBoardsBP == 3) {
59  nBoardsBPtot = 4;
60  if (!isBelowBeamPipe) {
61  --firstBoardId;
62  }
63  }
64 
65  for (int iboard = 0; iboard < nBoardsBPtot; ++iboard) {
66  int boardIdx = firstBoardId + iboard - 1;
67  if (nBoardsBP == 3) {
68  if ((isBelowBeamPipe && iboard == 3) || (!isBelowBeamPipe && iboard == 0)) {
69  boardIdx = 117;
70  }
71  }
72  columnStruct.boardsBP.push_back(boardIdx);
73  mBoardIndexes[boardIdx].deType = (boardIdx < 117) ? rpcType : 0;
74  mBoardIndexes[boardIdx].column = (boardIdx < 117) ? column : 0;
75  mBoardIndexes[boardIdx].line = (boardIdx < 117) ? iboard : 0;
76  }
77 }

235 std::vector<Mapping::MpStripIndex> Mapping::getNeighboursNBP(const Mapping::MpStripIndex& stripIndex, int rpcType) const
236 {
238  std::vector<MpStripIndex> neighbours;
239  MpStripIndex neigh;
240  for (int delta = -1; delta <= 1; delta += 2) {
241  neigh.column = stripIndex.column;
242  neigh.line = stripIndex.line;
243  neigh.strip = stripIndex.strip + delta;
244  if (neigh.strip < 0) {
245  if (stripIndex.column == 0 || (stripIndex.column == 1 && rpcType == 4)) {
246  continue;
247  }
248  --neigh.column;
249  if (!isValidLine(neigh.line, neigh.column, rpcType)) {
250  // This is useful for cut RPCs
251  continue;
252  }
253  neigh.strip = mDetectionElements[rpcType].columns[neigh.column].nStripsNBP - 1;
254  } else if (neigh.strip >= mDetectionElements[rpcType].columns[stripIndex.column].nStripsNBP) {
255  if (stripIndex.column == 6) {
256  continue;
257  }
258  ++neigh.column;
259  neigh.strip = 0;
260  }
261  neighbours.push_back(neigh);
262  }
263  return neighbours;
264 }

266 //______________________________________________________________________________
267 std::vector<Mapping::MpStripIndex> Mapping::getNeighboursBP(const Mapping::MpStripIndex& stripIndex, int rpcType) const
268 {
270  std::vector<MpStripIndex> neighbours;
271  MpStripIndex neigh;
272  // First search in the same column
273  for (int delta = -1; delta <= 1; delta += 2) {
274  neigh.column = stripIndex.column;
275  neigh.line = stripIndex.line;
276  neigh.strip = stripIndex.strip + delta;
277  if (neigh.strip < 0) {
278  --neigh.line;
279  if (!isValidLine(neigh.line, neigh.column, rpcType)) {
280  continue;
281  }
282  neigh.strip = 15;
283  } else if (neigh.strip >= 16) {
284  ++neigh.line;
285  if (!isValidLine(neigh.line, neigh.column, rpcType)) {
286  continue;
287  }
288  neigh.strip = 0;
289  }
290  neighbours.push_back(neigh);
291  }
292 
293  // Then search in the neighbour columns
294  int stripPitch = mDetectionElements[rpcType].columns[stripIndex.column].stripPitchBP;
295  for (int delta = -1; delta <= 1; delta += 2) {
296  neigh.column = stripIndex.column + delta;
297  if (!isValidColumn(neigh.column, rpcType)) {
298  continue;
299  }
300  int stripPitchNeigh = mDetectionElements[rpcType].columns[neigh.column].stripPitchBP;
301  int absStripNeigh = (16 * stripIndex.line + stripIndex.strip) * stripPitch / stripPitchNeigh;
302  neigh.line = absStripNeigh / 16;
303  if (!isValidLine(neigh.line, neigh.column, rpcType)) {
304  continue;
305  }
306  neigh.strip = absStripNeigh % 16;
307  neighbours.push_back(neigh);
308  if (stripPitch > stripPitchNeigh) {
309  neigh.strip = (neigh.strip % 2 == 0) ? neigh.strip + 1 : neigh.strip - 1;
310  neighbours.push_back(neigh);
311  }
312  }
313 
314  return neighbours;
315 }
