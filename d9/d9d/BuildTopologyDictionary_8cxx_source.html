27  void BuildTopologyDictionary::accountTopology(const ClusterTopology& cluster, float dX, float dZ)
28  {
29  mTotClusters++;
30 
31  // std::pair<unordered_map<unsigned long, std::pair<ClusterTopology,unsigned long>>::iterator,bool> ret;
32  auto ret = mTopologyMap.insert(std::make_pair(cluster.getHash(), std::make_pair(cluster, 1)));
33  if (ret.second == true) {
34  //___________________DEFINING_TOPOLOGY_CHARACTERISTICS__________________
35  TopologyInfo topInf;
36  int& rs = topInf.mSizeX = cluster.getRowSpan();
37  int& cs = topInf.mSizeZ = cluster.getColumnSpan();
38  //__________________COG_Deterrmination_____________
39  int tempyCOG = 0;
40  int tempzCOG = 0;
41  int tempFiredPixels = 0;
42  unsigned char tempChar = 0;
43  int s = 0;
44  int ic = 0;
45  int ir = 0;
46  for (unsigned int i = 2; i < cluster.getUsedBytes() + 2; i++) {
47  tempChar = cluster.getPattern()[i];
48  s = 128; // 0b10000000
49  while (s > 0) {
50  if ((tempChar & s) != 0) {
51  tempFiredPixels++;
52  tempyCOG += ir;
53  tempzCOG += ic;
54  }
55  ic++;
56  s /= 2;
57  if ((ir + 1) * ic == (rs * cs))
58  break;
59  if (ic == cs) {
60  ic = 0;
61  ir++;
62  }
63  }
64  if ((ir + 1) * ic == (rs * cs))
65  break;
66  }
67  topInf.mCOGx = 0.5 + (float)tempyCOG / (float)tempFiredPixels;
68  topInf.mCOGz = 0.5 + (float)tempzCOG / (float)tempFiredPixels;
69  topInf.mNpixels = tempFiredPixels;
70  topInf.mXmean = dX;
71  topInf.mXsigma2 = 0;
72  topInf.mZmean = dZ;
73  topInf.mZsigma2 = 0;
74  mMapInfo.insert(std::make_pair(cluster.getHash(), topInf));
75  } else {
76  int num = (ret.first->second.second++);
77  auto ind = mMapInfo.find(cluster.getHash());
78  float tmpxMean = ind->second.mXmean;
79  float newxMean = ind->second.mXmean = ((tmpxMean)*num + dX) / (num + 1);
80  float tmpxSigma2 = ind->second.mXsigma2;
81  ind->second.mXsigma2 =
82  (num * tmpxSigma2 + (dX - tmpxMean) * (dX - newxMean)) / (num + 1); // online variance algorithm
83  float tmpzMean = ind->second.mZmean;
84  float newzMean = ind->second.mZmean = ((tmpzMean)*num + dZ) / (num + 1);
85  float tmpzSigma2 = ind->second.mZsigma2;
86  ind->second.mZsigma2 =
87  (num * tmpzSigma2 + (dZ - tmpzMean) * (dZ - newzMean)) / (num + 1); // online variance algorithm
88  }
89  }

134  void BuildTopologyDictionary::setThresholdCumulative(double cumulative)
135  {
136  mTopologyFrequency.clear();
137  if (cumulative <= 0. || cumulative >= 1.)
138  cumulative = 0.99;
139  double totFreq = 0.;
140  for (auto&& p : mTopologyMap) {
141  mTopologyFrequency.push_back(std::make_pair(p.second.second, p.first));
142  }
143  std::sort(mTopologyFrequency.begin(), mTopologyFrequency.end(),
144  [](const std::pair<unsigned long, unsigned long>& couple1,
145  const std::pair<unsigned long, unsigned long>& couple2) { return (couple1.first > couple2.first); });
146  mNotInGroups = 0;
147  mNumberOfGroups = 0;
148  mDictionary.mFinalMap.clear();
149  for (auto& q : mTopologyFrequency) {
150  totFreq += ((double)(q.first)) / mTotClusters;
151  if (totFreq < cumulative) {
152  mNotInGroups++;
153  } else
154  break;
155  }
156  mFrequencyThreshold = ((double)(mTopologyFrequency[--mNotInGroups].first)) / mTotClusters;
157  while (((double)mTopologyFrequency[mNotInGroups].first) / mTotClusters == mFrequencyThreshold)
158  mNotInGroups--;
159  mFrequencyThreshold = ((double)mTopologyFrequency[mNotInGroups++].first) / mTotClusters;
160  mNumberOfGroups = mNotInGroups;
161  }

163  void BuildTopologyDictionary::groupRareTopologies()
164  {
165  std::cout << "groupRareTopologies: mTotClusters: " << mTotClusters << std::endl;
166 #ifdef _HISTO_
167  mHdist =
168  TH1F("mHdist", "Groups distribution",
169  mNumberOfGroups + TopologyDictionary::NumberOfRowClasses * TopologyDictionary::NumberOfColClasses, -0.5,
170  mNumberOfGroups + TopologyDictionary::NumberOfRowClasses * TopologyDictionary::NumberOfColClasses - 0.5);
171  mHdist.GetXaxis()->SetTitle("GroupID");
172  mHdist.SetFillColor(kRed);
173  mHdist.SetFillStyle(3005);
174 #endif
175 
176  double totFreq = 0.;
177  for (int j = 0; j < mNotInGroups; j++) {
178 #ifdef _HISTO_
179  mHdist.Fill(j, mTopologyFrequency[j].first);
180 #endif
181  totFreq += ((double)(mTopologyFrequency[j].first)) / mTotClusters;
182  GroupStruct gr;
183  gr.mHash = mTopologyFrequency[j].second;
184  gr.mFrequency = totFreq;
185  // rough estimation for the error considering a uniform distribution
186  gr.mErrX = std::sqrt(mMapInfo.find(gr.mHash)->second.mXsigma2);
187  gr.mErrZ = std::sqrt(mMapInfo.find(gr.mHash)->second.mZsigma2);
188  gr.mXCOG = mMapInfo.find(gr.mHash)->second.mCOGx;
189  gr.mZCOG = mMapInfo.find(gr.mHash)->second.mCOGz;
190  gr.mNpixels = mMapInfo.find(gr.mHash)->second.mNpixels;
191  mDictionary.mVectorOfGroupIDs.push_back(gr);
192  mDictionary.mFinalMap.insert(std::make_pair(gr.mHash, j));
193  }
194  // groupRareTopologies based on binning over number of rows and columns (TopologyDictionary::NumberOfRowClasses *
195  // NumberOfColClasse)
196  mNumberOfGroups += TopologyDictionary::NumberOfRowClasses * TopologyDictionary::NumberOfColClasses;
197  // array of groups
198  std::array<GroupStruct, TopologyDictionary::NumberOfRowClasses * TopologyDictionary::NumberOfColClasses> GroupArray;
199  std::array<unsigned long, TopologyDictionary::NumberOfRowClasses * TopologyDictionary::NumberOfColClasses>
200  groupCounts{ 0 };
201  auto func = [&GroupArray](int rowBinEdge, int colBinEdge, int& index) {
202  unsigned long provvHash = 0;
203  provvHash = (((unsigned long)(index + 1)) << 32) & 0xffffffff00000000;
204  GroupArray[index].mHash = provvHash;
205  GroupArray[index].mErrX = (rowBinEdge)*o2::ITSMFT::SegmentationAlpide::PitchRow / std::sqrt(12);
206  GroupArray[index].mErrZ = (colBinEdge)*o2::ITSMFT::SegmentationAlpide::PitchCol / std::sqrt(12);
207  GroupArray[index].mXCOG = rowBinEdge / 2;
208  GroupArray[index].mZCOG = colBinEdge / 2;
209  GroupArray[index].mNpixels = rowBinEdge * colBinEdge;
210  index++;
211  return;
212  };
213  int grNum = 0;
214  for (int ir = 0; ir < TopologyDictionary::NumberOfRowClasses - 1; ir++) {
215  for (int ic = 0; ic < TopologyDictionary::NumberOfColClasses - 1; ic++) {
216  func((ir + 1) * TopologyDictionary::RowClassSpan - 1, (ic + 1) * TopologyDictionary::ColClassSpan - 1, grNum);
217  }
218  func((ir + 1) * TopologyDictionary::RowClassSpan - 1, TopologyDictionary::MaxColSpan, grNum);
219  }
220  for (int ic = 0; ic < TopologyDictionary::NumberOfColClasses - 1; ic++) {
221  func(TopologyDictionary::MaxRowSpan, (ic + 1) * TopologyDictionary::ColClassSpan - 1, grNum);
222  unsigned long provvHash = 0;
223  }
224  func(TopologyDictionary::MaxRowSpan, TopologyDictionary::MaxColSpan, grNum);
225  if (grNum != TopologyDictionary::NumberOfColClasses * TopologyDictionary::NumberOfRowClasses) {
226  std::cout << "Wrong number of groups" << std::endl;
227  exit(1);
228  }
229  int rs;
230  int cs;
231  int index;
232 
233  for (unsigned int j = (unsigned int)mNotInGroups; j < mTopologyFrequency.size(); j++) {
234  unsigned long hash1 = mTopologyFrequency[j].second;
235  rs = mTopologyMap.find(hash1)->second.first.getRowSpan();
236  cs = mTopologyMap.find(hash1)->second.first.getColumnSpan();
237  index = (rs / TopologyDictionary::RowClassSpan) * TopologyDictionary::NumberOfRowClasses +
238  cs / TopologyDictionary::ColClassSpan;
239  if (index > TopologyDictionary::NumberOfRowClasses * TopologyDictionary::NumberOfColClasses - 1)
240  index = TopologyDictionary::NumberOfRowClasses * TopologyDictionary::NumberOfColClasses - 1;
241  groupCounts[index] += mTopologyFrequency[j].first;
242  }
243 
244  for (int i = 0; i < TopologyDictionary::NumberOfRowClasses * TopologyDictionary::NumberOfColClasses; i++) {
245  totFreq += ((double)groupCounts[i]) / mTotClusters;
246  GroupArray[i].mFrequency = totFreq;
247 #ifdef _HISTO_
248  mHdist.Fill(mNotInGroups + i, groupCounts[i]);
249 #endif
250  mDictionary.mVectorOfGroupIDs.push_back(GroupArray[i]);
251  }
252 #ifdef _HISTO_
253  mHdist.Scale(1. / mHdist.Integral());
254 #endif
255  }
