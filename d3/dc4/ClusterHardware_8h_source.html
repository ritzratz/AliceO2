24 struct ClusterHardware { // Draft of hardware clusters in bit-packed format.
25  // padPre: word 0, bits 0 - 18 (19 bit, two-complement, 4 fixed-point-bits) - Quantity needed to compute the pad.
26  // padPeak: word 0, bits 19-26 (8 bit, unsigned integer) - Offset of padPre.
27  // timePre: word 1, bits 0-18 (19 bit, two-complement, 4 fixed-point-bits) - Quantity needed to compute the time.
28  // timePeak: word 1, bits 19-27 (9 bit, unsigned integer) - Offset of timePre.
29  // sigmaPadPre: word 2, bits 0-19 (20 bit, unsigned, 4 fixed-point-bits) - Quantity needed to compute the sigma^2 of
30  // the pad.
31  // sigmaTimePre: word 3, bits 0-19 (20 bit, unsigned, 4 fixed-point-bits) - Quantity needed to compute the
32  // sigma^2 of the time.
33  // qMax: word 2, bits 20-30 (11 bit, 1 fixed-point-bit) - QMax of the cluster.
34  // qTot: word 4, bits 0-18 (19 bit, 4 fixed-point-bits) - Total charge of the cluster.
35  // row: word 3, bits 20-24 (5 bit, unsigned integer) - Row of the cluster (local, needs to add
36  // PadRegionInfo::getGlobalRowOffset)
37  // flags: word 4, bits 19-26 (8 bit) up to 8 bit flag field.
38  // remaining bits: reserved, must be 0!, could be used for additional bits later.
39  uint32_t word0; //< word0 of binary hardware cluster
40  uint32_t word1; //< word1 of binary hardware cluster
41  uint32_t word2; //< word2 of binary hardware cluster
42  uint32_t word3; //< word3 of binary hardware cluster
43  uint32_t word4; //< word4 of binary hardware cluster

107  void setCluster(float pad, float time, float sigmaPad2, float sigmaTime2, float qMax, float qTot, int row, int flags)
108  {
109  int max = qMax * 2.f;
110  int tot = qTot * 16.f;
111  qTot = tot / 16.f;
112  int padPeak = pad + 0.5;
113  float padOrg = pad;
114  pad -= padPeak;
115  int timePeak = time + 0.5;
116  time -= timePeak;
117  pad *= qTot;
118  time *= qTot;
119  int p = pad * 16.f;
120  int t = time * 16.f;
121  pad = p / 16.f;
122  time = t / 16.f;
123  sigmaPad2 = (sigmaPad2 + pad / qTot * pad / qTot) * qTot;
124  sigmaTime2 = (sigmaTime2 + time / qTot * time / qTot) * qTot;
125  int sp = sigmaPad2 * 16.f;
126  int st = sigmaTime2 * 16.f;
127  word0 = (p & 0x7FFFF) | ((padPeak & 0xFF) << 19);
128  word1 = (t & 0x7FFFF) | ((timePeak & 0x1FF) << 19);
129  word2 = (sp & 0xFFFFF) | ((max & 0x7FF) << 20);
130  word3 = (st & 0xFFFFF) | ((row & 0x1F) << 20);
131  word4 = (tot & 0x7FFFF) | ((flags & 0xFF) << 19);
132  }
