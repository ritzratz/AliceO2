37 bool Line::areParallel(const Line& firstLine, const Line& secondLine, const float precision)
38 {
39  float crossProdX{ firstLine.cosinesDirector[1] * secondLine.cosinesDirector[2] -
40  firstLine.cosinesDirector[2] * secondLine.cosinesDirector[1] };
41  float module{ std::abs(firstLine.cosinesDirector[1] * secondLine.cosinesDirector[2]) +
42  std::abs(firstLine.cosinesDirector[2] * secondLine.cosinesDirector[1]) };
43  if (std::abs(crossProdX) > precision * module)
44  return false;
45 
46  float crossProdY{ -firstLine.cosinesDirector[0] * secondLine.cosinesDirector[2] +
47  firstLine.cosinesDirector[2] * secondLine.cosinesDirector[0] };
48  module = std::abs(firstLine.cosinesDirector[0] * secondLine.cosinesDirector[2]) +
49  std::abs(firstLine.cosinesDirector[2] * secondLine.cosinesDirector[0]);
50  if (std::abs(crossProdY) > precision * module)
51  return false;
52 
53  float crossProdZ = firstLine.cosinesDirector[0] * secondLine.cosinesDirector[1] -
54  firstLine.cosinesDirector[1] * secondLine.cosinesDirector[0];
55  module = std::abs(firstLine.cosinesDirector[0] * secondLine.cosinesDirector[1]) +
56  std::abs(firstLine.cosinesDirector[1] * secondLine.cosinesDirector[0]);
57  if (std::abs(crossProdZ) > precision * module)
58  return false;
59 
60  return true;
61 }

63 float Line::getDCA(const Line& firstLine, const Line& secondLine, const float precision)
64 {
65  if (areParallel(firstLine, secondLine)) {
66  float squaredOriginsDistance{ 0 }, projectionOnFirstVector{ 0 }, normFirstVector{ 0 };
67  for (int i{ 0 }; i < 3; ++i) {
68  squaredOriginsDistance +=
69  (firstLine.originPoint[i] - secondLine.originPoint[i]) * (firstLine.originPoint[i] - secondLine.originPoint[i]);
70  projectionOnFirstVector += (firstLine.originPoint[i] - secondLine.originPoint[i]) * firstLine.cosinesDirector[i];
71  normFirstVector += firstLine.cosinesDirector[i] * firstLine.cosinesDirector[i];
72  }
73 
74  if (std::abs(normFirstVector) > precision) {
75  projectionOnFirstVector /= normFirstVector;
76  return std::sqrt(squaredOriginsDistance - projectionOnFirstVector * projectionOnFirstVector);
77  } else {
78  return -1;
79  }
80  } else {
81  std::array<float, 3> perpendicularVector{};
82  perpendicularVector[0] = firstLine.cosinesDirector[1] * secondLine.cosinesDirector[2] -
83  firstLine.cosinesDirector[2] * secondLine.cosinesDirector[1];
84  perpendicularVector[1] = -firstLine.cosinesDirector[0] * secondLine.cosinesDirector[2] +
85  firstLine.cosinesDirector[2] * secondLine.cosinesDirector[0];
86  perpendicularVector[2] = firstLine.cosinesDirector[0] * secondLine.cosinesDirector[1] -
87  firstLine.cosinesDirector[1] * secondLine.cosinesDirector[0];
88 
89  float normFirstVector{ 0 }, distance{ 0 };
90  for (int i{ 0 }; i < 3; ++i) {
91  normFirstVector += perpendicularVector[i] * perpendicularVector[i];
92  distance += (firstLine.originPoint[i] - secondLine.originPoint[i]) * perpendicularVector[i];
93  }
94  if (std::abs(normFirstVector) > precision) {
95  return std::abs(distance / std::sqrt(normFirstVector));
96  } else
97  return -1;
98  }
99 }
