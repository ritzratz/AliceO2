190 void someDataProducerAlgorithm(ProcessingContext& ctx)
191 {
192  size_t index = ctx.services().get<ParallelContext>().index1D();
193  sleep(1);
194  // Creates a new message of size collectionChunkSize which
195  // has "TPC" as data origin and "CLUSTERS" as data description.
196  auto tpcClusters = ctx.allocator().make<FakeCluster>(OutputSpec{ "TPC", "CLUSTERS", index }, collectionChunkSize);
197  int i = 0;
198 
199  for (auto& cluster : tpcClusters) {
200  assert(i < collectionChunkSize);
201  cluster.x = index;
202  cluster.y = i;
203  cluster.z = i;
204  cluster.q = rand() % 1000;
205  i++;
206  }
207 }

210 void someProcessingStageAlgorithm(ProcessingContext& ctx)
211 {
212  size_t index = ctx.services().get<ParallelContext>().index1D();
213 
214  const FakeCluster* inputDataTpc = reinterpret_cast<const FakeCluster*>(ctx.inputs().get("dataTPC").payload);
215 
216  auto processedTpcClusters = ctx.allocator().make<FakeCluster>(OutputSpec{ "TPC", "CLUSTERS_P", index },
217  collectionChunkSize);
218 
219  int i = 0;
220  for (auto& cluster : processedTpcClusters) {
221  assert(i < collectionChunkSize);
222  cluster.x = -inputDataTpc[i].x;
223  cluster.y = 2 * inputDataTpc[i].y;
224  cluster.z = inputDataTpc[i].z * inputDataTpc[i].q;
225  cluster.q = inputDataTpc[i].q;
226  i++;
227  }
