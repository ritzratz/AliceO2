519 inline void MatchTPCITS::removeTPCfromITS(int tpcMatchID, int itsMatchID)
520 {
522  auto& itsMatch = mMatchesITS[itsMatchID];
523  if (isValidatedITS(itsMatch))
524  return;
525  int topID = MinusOne, next = itsMatch.first;
526  while (next > MinusOne) {
527  auto& rcITS = mMatchRecordsITS[next];
528  if (rcITS.matchID == tpcMatchID) {
529  if (topID < 0) {
530  itsMatch.first = rcITS.nextRecID;
531  } else {
532  mMatchRecordsITS[topID].nextRecID = rcITS.nextRecID;
533  }
534  return;
535  }
536  topID = next;
537  next = rcITS.nextRecID;
538  }
539 }
