41 auto DataSampling::getEdgeMatcher(const QcTaskConfiguration& taskCfg)
42 {
43  return taskCfg.subSpec == -1 ?
44  [](const OutputSpec& externalOutput, const InputSpec& desiredData, const SubSpecificationType&) {
45  return externalOutput.origin == desiredData.origin &&
46  externalOutput.description == desiredData.description;
47  }
48  :
49  [](const OutputSpec& externalOutput, const InputSpec& desiredData,
50  const SubSpecificationType& desiredSubSpec) {
51  return externalOutput.origin == desiredData.origin &&
52  externalOutput.description == desiredData.description &&
53  externalOutput.subSpec == desiredSubSpec;
54  };
55 }

73 void DataSampling::GenerateInfrastructure(WorkflowSpec& workflow, const std::string& configurationSource)
74 {
75  QcTaskConfigurations tasks = readQcTasksConfiguration(configurationSource);
76  InfrastructureConfig infrastructureCfg = readInfrastructureConfiguration(configurationSource);
77 
78  for (auto&& task : tasks) {
79 
80  // if necessary, create FairMQ -> DPL proxies and after that, look for their outputs in workflow.
81  // (it is surely not an optimal way to do that, but this is only a temporary feature)
82  for (auto&& fairMqProxy : task.desiredFairMqData) {
83  workflow.emplace_back(specifyExternalFairMQDeviceProxy(
84  ("FairMQ_proxy_for_" + task.name).c_str(), Outputs{ fairMqProxy.outputSpec }, fairMqProxy.channelConfig.c_str(),
85  fairMqProxy.converterType == "o2DataModelAdaptor" ? o2DataModelAdaptor(fairMqProxy.outputSpec, 0, 1)
86  : incrementalConverter(fairMqProxy.outputSpec, 0, 1)));
87  }
88 
89  std::vector<std::unique_ptr<Dispatcher>> dispatchers;
90  auto areEdgesMatching = getEdgeMatcher(task);
91 
92  // Find all available outputs in workflow that match desired data. Create dispatchers that take them as inputs
93  // and provide them filtered as outputs.
94  for (auto&& desiredData : task.desiredDataSpecs) {
95  bool wasDataFound = false;
96  for (auto&& dataProcessor : workflow) {
97  for (auto&& externalOutput : dataProcessor.outputs) {
98  if (areEdgesMatching(externalOutput, desiredData, task.subSpec)) {
99 
100  wasDataFound = true;
101 
102  // if parallel dispatchers are not enabled, then edges will be added to the only one dispatcher.
103  // in other case, new dispatcher will be created for every parallel flow.
104  SubSpecificationType dispatcherSubSpec =
105  infrastructureCfg.enableParallelDispatchers ? externalOutput.subSpec : 0;
106 
107  auto res = std::find_if(dispatchers.begin(), dispatchers.end(), [dispatcherSubSpec](const auto& d) {
108  return d->getSubSpec() == dispatcherSubSpec;
109  });
110 
111  if (res != dispatchers.end()) {
112  (*res)->addSource(dataProcessor, externalOutput, desiredData.binding);
113  } else {
114  auto newDispatcher = createDispatcher(dispatcherSubSpec, task, infrastructureCfg);
115  newDispatcher->addSource(dataProcessor, externalOutput, desiredData.binding);
116  dispatchers.push_back(std::move(newDispatcher));
117  }
118  }
119  }
120  }
121  if (!wasDataFound) {
122  LOG(ERROR) << "Data '" << desiredData.binding << "' for QC task '" << task.name
123  << "' not found in given workflow";
124  }
125  }
126  for (auto& dispatcher : dispatchers) {
127  workflow.push_back(std::move(dispatcher->getDataProcessorSpec()));
128  }
129  }
130 }

134 QcTaskConfigurations DataSampling::readQcTasksConfiguration(const std::string& configurationSource)
135 {
136  QcTaskConfigurations tasks;
137  std::unique_ptr<ConfigurationInterface> configFile = ConfigurationFactory::getConfiguration(configurationSource);
138 
139  std::vector<std::string> taskNames;
140  try {
141  std::string taskNamesString = configFile->getString("DataSampling/tasksList").value();
142  boost::split(taskNames, taskNamesString, boost::is_any_of(","));
143  } catch (const boost::bad_optional_access&) {
144  LOG(ERROR) << "QC Task configuration error. In file " << configurationSource
145  << ", wrong or missing value DataSampling/tasksList";
146  return QcTaskConfigurations();
147  }
148 
149  for (auto&& taskName : taskNames) {
150 
151  QcTaskConfiguration task;
152  task.name = taskName;
153 
154  std::string taskInputsNames;
155  try {
156  std::string simpleQcTaskDefinition = configFile->getString(taskName + "/taskDefinition").value();
157  taskInputsNames = configFile->getString(simpleQcTaskDefinition + "/inputs").value();
158  task.fractionOfDataToSample = configFile->getFloat(simpleQcTaskDefinition + "/fraction").value();
159  if (task.fractionOfDataToSample <= 0 || task.fractionOfDataToSample > 1) {
160  LOG(ERROR) << "QC Task configuration error. In file " << configurationSource << ", value "
161  << simpleQcTaskDefinition + "/fraction"
162  << " is not in range (0,1]. Setting value to 0.";
163  task.fractionOfDataToSample = 0;
164  }
165  task.dispatcherType = configFile->getString(simpleQcTaskDefinition + "/dispatcherType").value_or("DPL");
166  // if there is a channelConfig specified, then user wants output in raw FairMQ layer, not DPL
167  task.fairMqOutputChannelConfig = configFile->getString(simpleQcTaskDefinition + "/channelConfig").value_or("");
168 
169  // FIXME: I do not like '-1' meaning 'all' - not 100% sure if it's safe to compare with '-1' later
170  task.subSpec = static_cast<header::DataHeader::SubSpecificationType>(
171  configFile->getInt(simpleQcTaskDefinition + "/subSpec").value_or(-1));
172 
173  } catch (const boost::bad_optional_access&) {
174  LOG(ERROR) << "QC Task configuration error. In file " << configurationSource
175  << ", missing value or values for task " << taskName;
176  continue;
177  }
178 
179  std::vector<std::string> taskInputsSplit;
180  boost::split(taskInputsSplit, taskInputsNames, boost::is_any_of(","));
181 
182  for (auto&& input : taskInputsSplit) {
183 
184  InputSpec desiredData;
185  try {
186  desiredData.binding = configFile->getString(input + "/inputName").value();
187 
188  std::string origin = configFile->getString(input + "/dataOrigin").value();
189  origin.copy(desiredData.origin.str, (size_t)desiredData.origin.size);
190 
191  std::string description = configFile->getString(input + "/dataDescription").value();
192  description.copy(desiredData.description.str, (size_t)desiredData.description.size);
193 
194  } catch (const boost::bad_optional_access&) {
195  LOG(ERROR) << "QC Task configuration error. In file " << configurationSource << " input " << input
196  << " has missing values";
197  continue;
198  }
199  task.desiredDataSpecs.push_back(desiredData);
200 
201  // for temporary feature
202  if (configFile->getInt(input + "/spawnConverter").value_or(0)) {
203  FairMqInput fairMqInput{ OutputSpec{
204  desiredData.origin, desiredData.description, task.subSpec == -1 ? 0 : task.subSpec,
205  },
206  configFile->getString(input + "/channelConfig").value_or(""),
207  configFile->getString(input + "/converterType").value_or("incrementalConverter") };
208  task.desiredFairMqData.push_back(fairMqInput);
209  }
210  }
211 
212  if (task.desiredDataSpecs.empty()) {
213  LOG(ERROR) << "QC Task configuration error. In file " << configurationSource << " task " << taskName
214  << " has no valid inputs";
215  continue;
216  }
217  tasks.push_back(task);
218  }
219 
220  return tasks;
221 }
