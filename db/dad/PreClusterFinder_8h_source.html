107 inline int PreClusterFinder::getNDEWithPreClusters(int& nUsedDigits)
108 {
110 
111  int nFiredDE(0);
112  nUsedDigits = 0;
113 
114  for (int iDE = 0; iDE < SNDEs; ++iDE) {
115 
116  DetectionElement& de(mDEs[iDE]);
117 
118  if (de.nOrderedPads[1] > 0) {
119  nUsedDigits += de.nOrderedPads[1];
120  ++nFiredDE;
121  }
122  }
123 
124  return nFiredDE;
125 }
