38 // A container to hold and manage MC truth information/labels.
39 // The actual MCtruth type is a generic template type and can be supplied by the user
40 // It is meant to manage associations from one "dataobject" identified by an index into an array
41 // to multiple TruthElements
42 
43 template <typename TruthElement>
44 class MCTruthContainer
45 {
46  private:
47  std::vector<MCTruthHeaderElement>
48  mHeaderArray; // the header structure array serves as an index into the actual storage
49  std::vector<TruthElement> mTruthArray; // the buffer containing the actual truth information

51  size_t getSize(int dataindex) const
52  {
53  // calculate size / number of labels from a difference in pointed indices
54  const auto size = (dataindex < mHeaderArray.size() - 1)
55  ? mHeaderArray[dataindex + 1].index - mHeaderArray[dataindex].index
56  : mTruthArray.size() - mHeaderArray[dataindex].index;
57  return size;
58  }

100  // add element for a particular dataindex
101  // at the moment only strictly consecutive modes are supported
102  void addElement(uint dataindex, TruthElement const& element)
103  {
104  if (dataindex < mHeaderArray.size()) {
105  // look if we have something for this dataindex already
106  // must currently be the last one
107  if (dataindex != (mHeaderArray.size() - 1)) {
108  throw std::runtime_error("MCTruthContainer: unsupported code path");
109  }
110  } else {
111  // assert(dataindex == mHeaderArray.size());
112 
113  // add empty holes
114  int holes = dataindex - mHeaderArray.size();
115  assert(holes >= 0);
116  for (int i = 0; i < holes; ++i) {
117  mHeaderArray.emplace_back(mTruthArray.size());
118  }
119  // add a new one
120  mHeaderArray.emplace_back(mTruthArray.size());
121  }
122  auto& header = mHeaderArray[dataindex];
123  mTruthArray.emplace_back(element);
124  }

126  // convenience interface to add multiple labels at once
127  // can use elements of any assignable type or sub-type
128  template <typename CompatibleLabel>
129  void addElements(uint dataindex, gsl::span<CompatibleLabel> elements)
130  {
131  static_assert(std::is_same<TruthElement, CompatibleLabel>::value ||
132  std::is_assignable<TruthElement, CompatibleLabel>::value ||
133  std::is_base_of<TruthElement, CompatibleLabel>::value,
134  "Need to add compatible labels");
135  for (auto& e : elements) {
136  addElement(dataindex, e);
137  }
138  }

148  // Add element at last position or for a previous index
149  // (at random access position).
150  // This might be a slow process since data has to be moved internally
151  // so this function should be used with care.
152  void addElementRandomAccess(uint dataindex, TruthElement const& element)
153  {
154  if (dataindex >= mHeaderArray.size()) {
155  // a new dataindex -> push element at back
156 
157  // we still forbid to leave holes
158  assert(dataindex == mHeaderArray.size());
159 
160  mHeaderArray.resize(dataindex + 1);
161  mHeaderArray[dataindex] = mTruthArray.size();
162  mTruthArray.emplace_back(element);
163  } else {
164  // if appending at end use fast function
165  if (dataindex == mHeaderArray.size() - 1) {
166  addElement(dataindex, element);
167  return;
168  }
169 
170  // existing dataindex
171  // have to:
172  // a) move data;
173  // b) insert new element;
174  // c) adjust indices of all headers right to this
175  auto currentindex = mHeaderArray[dataindex].index;
176  auto lastindex = currentindex + getSize(dataindex);
177  assert(currentindex >= 0);
178 
179  // resize truth array
180  mTruthArray.resize(mTruthArray.size() + 1);
181  // move data (have to do from right to left)
182  for (int i = mTruthArray.size() - 1; i > lastindex; --i) {
183  mTruthArray[i] = mTruthArray[i - 1];
184  }
185  // insert new element
186  mTruthArray[lastindex] = element;
187 
188  // fix headers
189  for (int i = dataindex + 1; i < mHeaderArray.size(); ++i) {
190  auto oldindex = mHeaderArray[i].index;
191  mHeaderArray[i].index = (oldindex != -1) ? oldindex + 1 : oldindex;
192  }
193  }
194  }
