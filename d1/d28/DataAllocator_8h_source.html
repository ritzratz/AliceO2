58  // In case no extra argument is provided and the passed type is trivially
59  // copyable and non polymorphic, the most likely wanted behavior is to create
60  // a message with that type, and so we do.
61  template <typename T>
62  typename std::enable_if<is_messageable<T>::value == true, T&>::type
63  make(const OutputSpec& spec)
64  {
65  DataChunk chunk = newChunk(spec, sizeof(T));
66  return *reinterpret_cast<T*>(chunk.data);
67  }

73  make(const OutputSpec& spec, size_t nElements)

174  template <typename W>
175  typename std::enable_if<is_specialization<W, ROOTSerialized>::value == true, void>::type
176  snapshot(const OutputSpec& spec, W wrapper)
177  {
178  using T = typename W::wrapped_type;
179  static_assert(std::is_same<typename W::hint_type, const char>::value || //
180  std::is_same<typename W::hint_type, TClass>::value || //
181  std::is_void<typename W::hint_type>::value, //
182  "class hint must be of type TClass or const char");
183 
184  FairMQMessagePtr payloadMessage(mDevice->NewMessage());
185  const TClass* cl = nullptr;
186  if (wrapper.getHint() == nullptr) {
187  // get TClass info by wrapped type
188  cl = TClass::GetClass(typeid(T));
189  } else if (std::is_same<typename W::hint_type, TClass>::value) {
190  // the class info has been passed directly
191  cl = reinterpret_cast<const TClass*>(wrapper.getHint());
192  } else if (std::is_same<typename W::hint_type, const char>::value) {
193  // get TClass info by optional name
194  cl = TClass::GetClass(reinterpret_cast<const char*>(wrapper.getHint()));
195  }
196  if (has_root_dictionary<T>::value == false && cl == nullptr) {
197  std::string msg("ROOT serialization not supported, dictionary not found for type ");
198  if (std::is_same<typename W::hint_type, const char>::value) {
199  msg += reinterpret_cast<const char*>(wrapper.getHint());
200  } else {
201  msg += typeid(T).name();
202  }
203  throw std::runtime_error(msg);
204  }
205  mDevice->Serialize<TMessageSerializer>(*payloadMessage, &wrapper(), cl);
206  addPartToContext(std::move(payloadMessage), spec, o2::header::gSerializationMethodROOT);
207  }

248  template <typename C>
249  typename std::enable_if<
250  is_specialization<C, std::vector>::value == true &&
251  std::is_pointer<typename C::value_type>::value == true &&
252  is_messageable<typename std::remove_pointer<typename C::value_type>::type>::value == true>::type
253  snapshot(const OutputSpec& spec, C const& v)

256  constexpr auto elementSizeInBytes = sizeof(ElementType);
257  auto sizeInBytes = elementSizeInBytes * v.size();
258  FairMQMessagePtr payloadMessage(mDevice->NewMessage(sizeInBytes));

261  for (auto const & pointer : v) {
