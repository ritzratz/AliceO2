148 inline void Cluster::setClusterUsage(Int_t n)
149 {
150  // set cluster usage counter
151  mNxNzN &= ~(kMaskClUse << kOffsClUse);
152  mNxNzN |= (n & kMaskClUse) << kOffsClUse;
153  if (n < 2)
154  resetBit(kShared);
155  if (!n)
156  resetBit(kUsed);
157 }
