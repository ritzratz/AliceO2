162 int createPipes(int maxFd, int* pipes)
163 {
164  auto p = pipe(pipes);
165  maxFd = maxFd > pipes[0] ? maxFd : pipes[0];
166  maxFd = maxFd > pipes[1] ? maxFd : pipes[1];
167 
168  if (p == -1) {
169  std::cerr << "Unable to create PIPE: ";
170  switch (errno) {
171  case EFAULT:
172  assert(false && "EFAULT while reading from pipe");
173  break;
174  case EMFILE:
175  std::cerr << "Too many active descriptors";
176  break;
177  case ENFILE:
178  std::cerr << "System file table is full";
179  break;
180  default:
181  std::cerr << "Unknown PIPE" << std::endl;
182  };
183  // Kill immediately both the parent and all the children
184  kill(-1 * getpid(), SIGKILL);
185  }
186  return maxFd;
187 }

201 void spawnDevice(DeviceSpec const& spec, std::map<int, size_t>& socket2DeviceInfo, DeviceControl& control,
202  DeviceExecution& execution, std::vector<DeviceInfo>& deviceInfos, int& maxFd, fd_set& childFdset)
203 {
204  int childstdout[2];
205  int childstderr[2];
206 
207  maxFd = createPipes(maxFd, childstdout);
208  maxFd = createPipes(maxFd, childstderr);
209 
210  // If we have a framework id, it means we have already been respawned
211  // and that we are in a child. If not, we need to fork and re-exec, adding
212  // the framework-id as one of the options.
213  pid_t id = 0;
214  id = fork();
215  // We are the child: prepare options and reexec.
216  if (id == 0) {
217  // We allow being debugged and do not terminate on SIGTRAP
218  signal(SIGTRAP, SIG_IGN);
219 
220  // We do not start the process if control.noStart is set.
221  if (control.stopped) {
222  kill(getpid(), SIGSTOP);
223  }
224 
225  // This is the child. We close the read part of the pipe, stdout
226  // and dup2 the write part of the pipe on it. Then we can restart.
227  close(childstdout[0]);
228  close(childstderr[0]);
229  close(STDOUT_FILENO);
230  close(STDERR_FILENO);
231  dup2(childstdout[1], STDOUT_FILENO);
232  dup2(childstderr[1], STDERR_FILENO);
233  execvp(execution.args[0], execution.args.data());
234  }
235 
236  // This is the parent. We close the write end of
237  // the child pipe and and keep track of the fd so
238  // that we can later select on it.
239  struct sigaction sa_handle_int;
240  sa_handle_int.sa_handler = handle_sigint;
241  sigemptyset(&sa_handle_int.sa_mask);
242  sa_handle_int.sa_flags = SA_RESTART;
243  if (sigaction(SIGINT, &sa_handle_int, nullptr) == -1) {
244  perror("Unable to install signal handler");
245  exit(1);
246  }
247 
248  std::cout << "Starting " << spec.id << " on pid " << id << "\n";
249  DeviceInfo info;
250  info.pid = id;
251  info.active = true;
252  info.readyToQuit = false;
253  info.historySize = 1000;
254  info.historyPos = 0;
255  info.maxLogLevel = LogParsingHelpers::LogLevel::Debug;
256 
257  socket2DeviceInfo.insert(std::make_pair(childstdout[0], deviceInfos.size()));
258  socket2DeviceInfo.insert(std::make_pair(childstderr[0], deviceInfos.size()));
259  deviceInfos.emplace_back(info);
260  // Let's add also metrics information for the given device
261  gDeviceMetricsInfos.emplace_back(DeviceMetricsInfo{});
262 
263  close(childstdout[1]);
264  close(childstderr[1]);
265  FD_SET(childstdout[0], &childFdset);
266  FD_SET(childstderr[0], &childFdset);
267 }

269 void processChildrenOutput(DriverInfo& driverInfo, DeviceInfos& infos, DeviceSpecs const& specs,
270  DeviceControls& controls, std::vector<DeviceMetricsInfo>& metricsInfos)
271 {
272  // Wait for children to say something. When they do
273  // print it.
274  fd_set fdset;
275  timeval timeout;
276  timeout.tv_sec = 0;
277  timeout.tv_usec = 16666; // This should be enough to allow 60 HZ redrawing.
278  memcpy(&fdset, &driverInfo.childFdset, sizeof(fd_set));
279  int numFd = select(driverInfo.maxFd, &fdset, nullptr, nullptr, &timeout);
280  if (numFd == 0) {
281  return;
282  }
283  for (int si = 0; si < driverInfo.maxFd; ++si) {
284  if (FD_ISSET(si, &fdset)) {
285  assert(driverInfo.socket2DeviceInfo.find(si) != driverInfo.socket2DeviceInfo.end());
286  auto& info = infos[driverInfo.socket2DeviceInfo[si]];
287 
288  bool fdActive = getChildData(si, info);
289  // If the pipe was closed due to the process exiting, we
290  // can avoid the select.
291  if (!fdActive) {
292  info.active = false;
293  close(si);
294  FD_CLR(si, &driverInfo.childFdset);
295  }
296  --numFd;
297  }
298  // FIXME: no need to check after numFd gets to 0.
299  }
300  // Display part. All you need to display should actually be in
301  // `infos`.
302  // TODO: split at \n
303  // TODO: update this only once per 1/60 of a second or
304  // things like this.
305  // TODO: have multiple display modes
306  // TODO: graphical view of the processing?
307  assert(infos.size() == controls.size());
308  std::smatch match;
309  std::string token;
310  const std::string delimiter("\n");
311  for (size_t di = 0, de = infos.size(); di < de; ++di) {
312  DeviceInfo& info = infos[di];
313  DeviceControl& control = controls[di];
314  DeviceMetricsInfo& metrics = metricsInfos[di];
315 
316  if (info.unprinted.empty()) {
317  continue;
318  }
319 
320  auto s = info.unprinted;
321  size_t pos = 0;
322  info.history.resize(info.historySize);
323  info.historyLevel.resize(info.historySize);
324 
325  while ((pos = s.find(delimiter)) != std::string::npos) {
326  token = s.substr(0, pos);
327  auto logLevel = LogParsingHelpers::parseTokenLevel(token);
328 
329  // Check if the token is a metric from SimpleMetricsService
330  // if yes, we do not print it out and simply store it to be displayed
331  // in the GUI.
332  // Then we check if it is part of our Poor man control system
333  // if yes, we execute the associated command.
334  if (parseMetric(token, match)) {
335  LOG(DEBUG) << "Found metric with key " << match[2] << " and value " << match[4];
336  processMetric(match, metrics);
337  } else if (parseControl(token, match)) {
338  auto command = match[1];
339  auto validFor = match[2];
340  LOG(DEBUG) << "Found control command " << command << " from pid " << info.pid << " valid for " << validFor;
341  if (command == "QUIT") {
342  if (validFor == "ALL") {
343  for (auto& deviceInfo : infos) {
344  deviceInfo.readyToQuit = true;
345  }
346  }
347  }
348  } else if (!control.quiet && (strstr(token.c_str(), control.logFilter) != nullptr) &&
349  logLevel >= control.logLevel) {
350  assert(info.historyPos >= 0);
351  assert(info.historyPos < info.history.size());
352  info.history[info.historyPos] = token;
353  info.historyLevel[info.historyPos] = logLevel;
354  info.historyPos = (info.historyPos + 1) % info.history.size();
355  std::cout << "[" << info.pid << "]: " << token << std::endl;
356  }
357  // We keep track of the maximum log error a
358  // device has seen.
359  if (logLevel > info.maxLogLevel && logLevel > LogParsingHelpers::LogLevel::Info &&
360  logLevel != LogParsingHelpers::LogLevel::Unknown) {
361  info.maxLogLevel = logLevel;
362  }
363  if (logLevel == LogParsingHelpers::LogLevel::Error) {
364  info.lastError = token;
365  }
366  s.erase(0, pos + delimiter.length());
367  }
368  info.unprinted = s;
369  }
370  // FIXME: for the gui to work correctly I would actually need to
371  // run the loop more often and update whenever enough time has
372  // passed.
373 }

448  // Populate options from the command line. Notice that only the options

451  boost::program_options::options_description optsDesc;

461  serviceRegistry.registerService<ControlService>(new TextControlService());

465  serviceRegistry.registerService<RawDeviceService>(new SimpleRawDeviceService(nullptr));
466 
467  if (spec.inputs.empty()) {
468  LOG(DEBUG) << spec.id << " is a source\n";
469  device.reset(new DataSourceDevice(spec, serviceRegistry));
470  } else {
471  LOG(DEBUG) << spec.id << " is a processor\n";
472  device.reset(new DataProcessingDevice(spec, serviceRegistry));

476 
477  runner.AddHook<fair::mq::hooks::InstantiateDevice>([&device](fair::mq::DeviceRunner& r) {
478  r.fDevice = std::shared_ptr<FairMQDevice>{ std::move(device) };
479  TurnOffColors::apply(false, 0);
480  });
481 
482  return runner.Run();
483  } catch (std::exception& e) {
484  LOG(ERROR) << "Unhandled exception reached the top of main: " << e.what() << ", device shutting down.";
485  return 1;
486  } catch (...) {
487  LOG(ERROR) << "Unknown exception reached the top of main.\n";
488  return 1;
489  }
490  return 0;
491 }

502 // This is the handler for the parent inner loop.
503 int runStateMachine(DataProcessorSpecs const& workflow, DriverControl& driverControl, DriverInfo& driverInfo,
504  std::vector<DeviceMetricsInfo>& metricsInfos, std::string frameworkId)
505 {
506  DeviceSpecs deviceSpecs;
507  DeviceInfos infos;
508  DeviceControls controls;
509  DeviceExecutions deviceExecutions;
510 
511  void* window = nullptr;
512  decltype(getGUIDebugger(infos, deviceSpecs, metricsInfos, driverInfo, controls, driverControl)) debugGUICallback;
513 
514  if (driverInfo.batch == false) {
515  window = initGUI("O2 Framework debug GUI");
516  }
517  if (driverInfo.batch == false && window == nullptr) {
518  LOG(WARN) << "Could not create GUI. Switching to batch mode. Do you have GLFW on your system?";
519  driverInfo.batch = true;
520  }
521  bool guiQuitRequested = false;
522 
523  // FIXME: I should really have some way of exiting the
524  // parent..
525  DriverState current;
526  while (true) {
527  // If control forced some transition on us, we push it to the queue.
528  if (driverControl.forcedTransitions.empty() == false) {
529  for (auto transition : driverControl.forcedTransitions) {
530  driverInfo.states.push_back(transition);
531  }
532  driverControl.forcedTransitions.resize(0);
533  }
534  // In case a timeout was requested, we check if we are running
535  // for more than the timeout duration and exit in case that's the case.
536  {
537  auto currentTime = std::chrono::steady_clock::now();
538  std::chrono::duration<double> diff = currentTime - driverInfo.startTime;
539  if ((graceful_exit == false) && (driverInfo.timeout > 0) && (diff.count() > driverInfo.timeout)) {
540  LOG(INFO) << "Timout ellapsed. Requesting to quit.";
541  graceful_exit = true;
542  }
543  }
544  // Move to exit loop if sigint was sent we execute this only once.
545  if (graceful_exit == true && driverInfo.sigintRequested == false) {
546  driverInfo.sigintRequested = true;
547  driverInfo.states.resize(0);
548  driverInfo.states.push_back(DriverState::QUIT_REQUESTED);
549  driverInfo.states.push_back(DriverState::GUI);
550  }
551  // If one of the children dies and sigint was not requested
552  // we should decide what to do.
553  if (sigchld_requested == true && driverInfo.sigchldRequested == false) {
554  driverInfo.sigchldRequested = true;
555  pruneGUI(driverInfo.states);
556  driverInfo.states.push_back(DriverState::HANDLE_CHILDREN);
557  driverInfo.states.push_back(DriverState::GUI);
558  }
559  if (driverInfo.states.empty() == false) {
560  current = driverInfo.states.back();
561  } else {
562  current = DriverState::UNKNOWN;
563  }
564  driverInfo.states.pop_back();
565  switch (current) {
566  case DriverState::INIT:
567  LOG(INFO) << "Initialising O2 Data Processing Layer";
568 
569  // Install signal handler for quitting children.
570  driverInfo.sa_handle_child.sa_handler = &handle_sigchld;
571  sigemptyset(&driverInfo.sa_handle_child.sa_mask);
572  driverInfo.sa_handle_child.sa_flags = SA_RESTART | SA_NOCLDSTOP;
573  if (sigaction(SIGCHLD, &driverInfo.sa_handle_child, nullptr) == -1) {
574  perror(nullptr);
575  exit(1);
576  }
577  FD_ZERO(&(driverInfo.childFdset));
578 
583  driverInfo.states.push_back(DriverState::RUNNING);
584  driverInfo.states.push_back(DriverState::GUI);
585  // driverInfo.states.push_back(DriverState::REDEPLOY_GUI);
586  LOG(INFO) << "O2 Data Processing Layer initialised. We brake for nobody.";
587  break;
588  case DriverState::MATERIALISE_WORKFLOW:
589  try {
590  DeviceSpecHelpers::dataProcessorSpecs2DeviceSpecs(workflow, driverInfo.channelPolicies, deviceSpecs);
591  // This should expand nodes so that we can build a consistent DAG.
592  } catch (std::runtime_error& e) {
593  std::cerr << "Invalid workflow: " << e.what() << std::endl;
594  return 1;
595  }
596  break;
597  case DriverState::DO_CHILD:
598  for (auto& spec : deviceSpecs) {
599  if (spec.id == frameworkId) {
600  return doChild(driverInfo.argc, driverInfo.argv, spec);
601  }
602  }
603  LOG(ERROR) << "Unable to find component with id " << frameworkId;
604  break;
605  case DriverState::REDEPLOY_GUI:
606  // The callback for the GUI needs to be recalculated every time
607  // the deployed configuration changes, e.g. a new device
608  // has been added to the topology.
609  // We need to recreate the GUI callback every time we reschedule
610  // because getGUIDebugger actually recreates the GUI state.
611  if (window) {
612  debugGUICallback = getGUIDebugger(infos, deviceSpecs, metricsInfos, driverInfo, controls, driverControl);
613  }
614  break;
615  case DriverState::SCHEDULE:
616  // FIXME: for the moment modifying the topology means we rebuild completely
617  // all the devices and we restart them. This is also what DDS does at
618  // a larger scale. In principle one could try to do a delta and only
619  // restart the data processors which need to be restarted.
620  LOG(INFO) << "Redeployment of configuration asked.";
621  controls.resize(deviceSpecs.size());
622  deviceExecutions.resize(deviceSpecs.size());
623 
624  DeviceSpecHelpers::prepareArguments(driverInfo.argc, driverInfo.argv, driverControl.defaultQuiet,
625  driverControl.defaultStopped, deviceSpecs, deviceExecutions, controls);
626  for (size_t di = 0; di < deviceSpecs.size(); ++di) {
627  spawnDevice(deviceSpecs[di], driverInfo.socket2DeviceInfo, controls[di], deviceExecutions[di], infos,
628  driverInfo.maxFd, driverInfo.childFdset);
629  }
630  driverInfo.maxFd += 1;
631  assert(infos.empty() == false);
632  LOG(INFO) << "Redeployment of configuration done.";
633  break;
634  case DriverState::RUNNING:
635  // Calculate what we should do next and eventually
636  // show the GUI
637  if (driverInfo.batch || guiQuitRequested || (checkIfCanExit(infos) == true)) {
638  // Something requested to quit. Let's update the GUI
639  // one more time and then EXIT.
640  LOG(INFO) << "Quitting";
641  driverInfo.states.push_back(DriverState::QUIT_REQUESTED);
642  driverInfo.states.push_back(DriverState::GUI);
643  } else if (infos.size() != deviceSpecs.size()) {
644  // If the number of deviceSpecs is different from
645  // the DeviceInfos it means the speicification
646  // does not match what is running, so we need to do
647  // further scheduling.
648  driverInfo.states.push_back(DriverState::RUNNING);
649  driverInfo.states.push_back(DriverState::GUI);
650  driverInfo.states.push_back(DriverState::REDEPLOY_GUI);
651  driverInfo.states.push_back(DriverState::GUI);
652  driverInfo.states.push_back(DriverState::SCHEDULE);
653  driverInfo.states.push_back(DriverState::GUI);
654  } else if (deviceSpecs.size() == 0) {
655  LOG(INFO) << "No device resulting from the workflow. Quitting.";
656  // If there are no deviceSpecs, we exit.
657  driverInfo.states.push_back(DriverState::EXIT);
658  } else {
659  driverInfo.states.push_back(DriverState::RUNNING);
660  driverInfo.states.push_back(DriverState::GUI);
661  }
662  processChildrenOutput(driverInfo, infos, deviceSpecs, controls, metricsInfos);
663  break;
664  case DriverState::GUI:
665  if (window) {
666  guiQuitRequested = (pollGUI(window, debugGUICallback) == false);
667  }
668  break;
669  case DriverState::QUIT_REQUESTED:
670  LOG(INFO) << "QUIT_REQUESTED" << std::endl;
671  guiQuitRequested = true;
672  killChildren(infos, SIGTERM);
673  driverInfo.states.push_back(DriverState::HANDLE_CHILDREN);
674  driverInfo.states.push_back(DriverState::GUI);
675  break;
676  case DriverState::HANDLE_CHILDREN:
677  // I allow queueing of more sigchld only when
678  // I process the previous call
679  sigchld_requested = false;
680  driverInfo.sigchldRequested = false;
681  processChildrenOutput(driverInfo, infos, deviceSpecs, controls, metricsInfos);
682  processSigChild(infos);
683  if (areAllChildrenGone(infos) == true &&
684  (guiQuitRequested || (checkIfCanExit(infos) == true) || graceful_exit)) {
685  // We move to the exit, regardless of where we were
686  driverInfo.states.resize(0);
687  driverInfo.states.push_back(DriverState::EXIT);
688  driverInfo.states.push_back(DriverState::GUI);
689  } else if (areAllChildrenGone(infos) == false &&
690  (guiQuitRequested || checkIfCanExit(infos) == true || graceful_exit)) {
691  driverInfo.states.push_back(DriverState::HANDLE_CHILDREN);
692  driverInfo.states.push_back(DriverState::GUI);
693  } else {
694  driverInfo.states.push_back(DriverState::GUI);
695  }
696  break;
697  case DriverState::EXIT:
698  return calculateExitCode(infos);
699  case DriverState::PERFORM_CALLBACKS:
700  for (auto& callback : driverControl.callbacks) {
701  callback(deviceSpecs, deviceExecutions);
702  }
703  driverControl.callbacks.clear();
704  driverInfo.states.push_back(DriverState::GUI);
705  break;
706  default:
707  LOG(ERROR) << "Driver transitioned in an unknown state. Shutting down.";
708  driverInfo.states.push_back(DriverState::QUIT_REQUESTED);
709  }
710  }
