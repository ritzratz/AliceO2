54 
55  {
56  std::unique_ptr<TFile> testFile(TFile::Open(fileName.c_str(), "RECREATE"));
57  std::unique_ptr<TTree> testTree = std::make_unique<TTree>("testtree", "testtree");
58 
59  std::vector<o2::test::Polymorphic> valarray;
60  auto* branch = testTree->Branch("dataarray", &valarray);
61 
62  for (int entry = 0; entry < kTreeSize; entry++) {
63  valarray.clear();

65  valarray.emplace_back((entry * 10) + idx);
66  }

93 DataProcessorSpec getSinkSpec()
94 {
95  auto processingFct = [](ProcessingContext& pc) {
96  static int counter = 0;
97  using DataHeader = o2::header::DataHeader;
98  for (auto& input : pc.inputs()) {
99  auto dh = o2::header::get<const DataHeader>(input.header);
100  LOG(INFO) << dh->dataOrigin.str << " " << dh->dataDescription.str << " " << dh->payloadSize;
101  }
102  auto data = pc.inputs().get<std::vector<o2::test::Polymorphic>>("input");
103 
104  LOG(INFO) << "count: " << counter << " data elements:" << data->size();
105  ASSERT_ERROR(counter + 1 == data->size());
106  for (int idx = 0; idx < data->size(); idx++) {
107  LOG(INFO) << (*data)[idx].get();
108  ASSERT_ERROR((*data)[idx].get() == 10 * counter + idx);
109  }
110  if (++counter >= kTreeSize) {
111  pc.services().get<ControlService>().readyToQuit(true);
112  }
113  };
114 
115  return DataProcessorSpec{ "sink", // name of the processor
116  { InputSpec{ "input", "TST", "ARRAYOFDATA", 0, InputSpec::Timeframe } },
117  Outputs{},
118  AlgorithmSpec(processingFct) };
