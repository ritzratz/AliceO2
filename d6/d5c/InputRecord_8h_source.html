193  // substitution for messageable objects with ROOT dictionary
194  // the operation depends on the transmitted serialization method
195  template <typename T>
196  typename std::enable_if<is_messageable<T>::value &&
197  std::is_same<T, DataRef>::value == false &&
198  has_root_dictionary<T>::value,
199  std::unique_ptr<T const, Deleter<T const>>>::type
200  get(char const* binding) const
201  {
202  using DataHeader = o2::header::DataHeader;
203 
204  auto ref = this->get(binding);
205  auto header = o2::header::get<const DataHeader>(ref.header);
206  auto method = header->payloadSerializationMethod;
207  if (method == o2::header::gSerializationMethodNone) {
208  auto const* ptr = reinterpret_cast<T const*>(ref.payload);
209  // return type with non-owning Deleter instance
210  std::unique_ptr<T const, Deleter<T const>> result(ptr, Deleter<T const>(false));
211  return std::move(result);
212  } else if (method == o2::header::gSerializationMethodROOT) {
213  // explicitely specify serialization method to ROOT-serialized because type T
214  // is messageable and a different method would be deduced in DataRefUtils
215  // return type with owning Deleter instance, forwarding to default_deleter
216  std::unique_ptr<T const, Deleter<T const>> result(DataRefUtils::as<ROOTSerialized<T>>(ref).release());
217  return std::move(result);
218  } else {
219  throw std::runtime_error("Attempt to extract object from message with unsupported serializayion type");
220  }
221  }
222
