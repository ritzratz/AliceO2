181 bool areClustersEqual(Cluster2D& cl1, Cluster2D& cl2)
182 {
183  int nBad = 0;
184  float precision = 1.e-3;
185  if (cl1.deId != cl2.deId) {
186  std::cerr << "Id: " << cl1.deId << " != " << cl2.deId << std::endl;
187  ++nBad;
188  }
189 
190  if (std::abs(cl1.xCoor - cl2.xCoor) > precision) {
191  std::cerr << "xCoor: " << cl1.xCoor << " != " << cl2.xCoor << std::endl;
192  ++nBad;
193  }
194 
195  if (std::abs(cl1.yCoor - cl2.yCoor) > precision) {
196  std::cerr << "yCoor: " << cl1.yCoor << " != " << cl2.yCoor << std::endl;
197  ++nBad;
198  }
199 
200  if (std::abs(cl1.sigmaX2 - cl2.sigmaX2) > precision) {
201  std::cerr << "sigmaX2: " << cl1.sigmaX2 << " != " << cl2.sigmaX2 << std::endl;
202  ++nBad;
203  }
204 
205  if (std::abs(cl1.sigmaY2 - cl2.sigmaY2) > precision) {
206  std::cerr << "sigmaY2: " << cl1.sigmaY2 << " != " << cl2.sigmaY2 << std::endl;
207  ++nBad;
208  }
209 
210  return (nBad == 0);
211 }

242 bool isWithinUncertainties(float xPos, float yPos, Cluster2D& cl)
243 {
244  std::string str[2] = { "x", "y" };
245  float inputPos[2] = { xPos, yPos };
246  float recoPos[2] = { cl.xCoor, cl.yCoor };
247  float sigma2[2] = { cl.sigmaX2, cl.sigmaY2 };
248  bool isOk = true;
249  for (int icoor = 0; icoor < 2; ++icoor) {
250  float sigma = std::sqrt(sigma2[icoor]);
251  if (sigma > 5. || std::abs(recoPos[icoor] - inputPos[icoor]) > 5 * sigma) {
252  std::cerr << str[icoor] << " position input " << inputPos[icoor] << " reco " << recoPos[icoor] << " sigma "
253  << sigma << std::endl;
254  isOk = false;
255  }
256  }
257  return isOk;
258 }

260 std::vector<ColumnData> getFiredStrips(float xPos, float yPos, int deId, Mapping& mapping)
261 {
262  // This is a quite simple case just for testing purposes.
263  // The fired strips are simply the fired stripp itself + its neighbours.
264  // However, in the bending plane, this also consists of strips with no overlap
265  // with the non-bending plane
266  std::vector<ColumnData> columns;
267  for (int cathode = 1; cathode >= 0; --cathode) {
268  Mapping::MpStripIndex stripIndex = mapping.stripByPosition(xPos, yPos, cathode, deId, false);
269  if (!stripIndex.isValid()) {
270  continue;
271  }
272  std::vector<Mapping::MpStripIndex> neighbours = mapping.getNeighbours(stripIndex, cathode, deId);
273  neighbours.push_back(stripIndex);
274  for (auto& neigh : neighbours) {
275  ColumnData* columnStruct = nullptr;
276  for (auto& currCol : columns) {
277  if (currCol.columnId == neigh.column) {
278  columnStruct = &currCol;
279  break;
280  }
281  }
282  if (!columnStruct) {
283  if (cathode == 0) {
284  // For the sake of simplicity we reject the neighbour bending strips
285  // that do not overlap to the non-bending plane
286  continue;
287  }
288  columns.emplace_back(ColumnData());
289  columnStruct = &columns.back();
290  columnStruct->deId = deId;
291  columnStruct->columnId = neigh.column;
292  }
293  int line = (cathode == 1) ? 4 : neigh.line;
294  columnStruct->patterns.patterns[line] |= (1 << neigh.strip);
295  }
296  }
297  return columns;
298 }
