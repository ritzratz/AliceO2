37 void TrackerTraits<false>::computeLayerTracklets(PrimaryVertexContext& primaryVertexContext)
38 {
39  for (int iLayer{ 0 }; iLayer < Constants::ITS::TrackletsPerRoad; ++iLayer) {
40  if (primaryVertexContext.getClusters()[iLayer].empty() || primaryVertexContext.getClusters()[iLayer + 1].empty()) {
41  return;
42  }
43 
44  const float3& primaryVertex = primaryVertexContext.getPrimaryVertex();
45  const int currentLayerClustersNum{ static_cast<int>(primaryVertexContext.getClusters()[iLayer].size()) };
46 
47  for (int iCluster{ 0 }; iCluster < currentLayerClustersNum; ++iCluster) {
48  const Cluster& currentCluster{ primaryVertexContext.getClusters()[iLayer][iCluster] };
49 
50  const float tanLambda{ (currentCluster.zCoordinate - primaryVertex.z) / currentCluster.rCoordinate };
51  const float directionZIntersection{ tanLambda * (Constants::ITS::LayersRCoordinate()[iLayer + 1] -
52  currentCluster.rCoordinate) +
53  currentCluster.zCoordinate };
54 
55  const int4 selectedBinsRect{ TrackingUtils::getBinsRect(currentCluster, iLayer, directionZIntersection) };
56 
57  if (selectedBinsRect.x == 0 && selectedBinsRect.y == 0 && selectedBinsRect.z == 0 && selectedBinsRect.w == 0) {
58  continue;
59  }
60 
61  int phiBinsNum{ selectedBinsRect.w - selectedBinsRect.y + 1 };
62 
63  if (phiBinsNum < 0) {
64  phiBinsNum += Constants::IndexTable::PhiBins;
65  }
66 
67  for (int iPhiBin{ selectedBinsRect.y }, iPhiCount{ 0 }; iPhiCount < phiBinsNum;
68  iPhiBin = ++iPhiBin == Constants::IndexTable::PhiBins ? 0 : iPhiBin, iPhiCount++) {
69 
70  const int firstBinIndex{ IndexTableUtils::getBinIndex(selectedBinsRect.x, iPhiBin) };
71  const int maxBinIndex{ firstBinIndex + selectedBinsRect.z - selectedBinsRect.x + 1 };
72  const int firstRowClusterIndex = primaryVertexContext.getIndexTables()[iLayer][firstBinIndex];
73  const int maxRowClusterIndex = primaryVertexContext.getIndexTables()[iLayer][maxBinIndex];
74 
75  for (int iNextLayerCluster{ firstRowClusterIndex }; iNextLayerCluster < maxRowClusterIndex;
76  ++iNextLayerCluster) {
77 
78  const Cluster& nextCluster{ primaryVertexContext.getClusters()[iLayer + 1][iNextLayerCluster] };
79 
80  const float deltaZ{ MATH_ABS(tanLambda * (nextCluster.rCoordinate - currentCluster.rCoordinate) +
81  currentCluster.zCoordinate - nextCluster.zCoordinate) };
82  const float deltaPhi{ MATH_ABS(currentCluster.phiCoordinate - nextCluster.phiCoordinate) };
83 
84  if (deltaZ < Constants::Thresholds::TrackletMaxDeltaZThreshold()[iLayer] &&
85  (deltaPhi < Constants::Thresholds::PhiCoordinateCut ||
86  MATH_ABS(deltaPhi - Constants::Math::TwoPi) < Constants::Thresholds::PhiCoordinateCut)) {
87 
88  if (iLayer > 0 &&
89  primaryVertexContext.getTrackletsLookupTable()[iLayer - 1][iCluster] == Constants::ITS::UnusedIndex) {
90 
91  primaryVertexContext.getTrackletsLookupTable()[iLayer - 1][iCluster] =
92  primaryVertexContext.getTracklets()[iLayer].size();
93  }
94 
95  primaryVertexContext.getTracklets()[iLayer].emplace_back(iCluster, iNextLayerCluster, currentCluster,
96  nextCluster);
97  }
98  }
99  }
100  }
101  }

105 void TrackerTraits<false>::computeLayerCells(PrimaryVertexContext& primaryVertexContext)
106 {
107  for (int iLayer{ 0 }; iLayer < Constants::ITS::CellsPerRoad; ++iLayer) {
108 
109  if (primaryVertexContext.getTracklets()[iLayer + 1].empty() ||
110  primaryVertexContext.getTracklets()[iLayer].empty()) {
111 
112  return;
113  }
114 
115  const float3& primaryVertex = primaryVertexContext.getPrimaryVertex();
116  const int currentLayerTrackletsNum{ static_cast<int>(primaryVertexContext.getTracklets()[iLayer].size()) };
117 
118  for (int iTracklet{ 0 }; iTracklet < currentLayerTrackletsNum; ++iTracklet) {
119 
120  const Tracklet& currentTracklet{ primaryVertexContext.getTracklets()[iLayer][iTracklet] };
121  const int nextLayerClusterIndex{ currentTracklet.secondClusterIndex };
122  const int nextLayerFirstTrackletIndex{
123  primaryVertexContext.getTrackletsLookupTable()[iLayer][nextLayerClusterIndex]
124  };
125 
126  if (nextLayerFirstTrackletIndex == Constants::ITS::UnusedIndex) {
127 
128  continue;
129  }
130 
131  const Cluster& firstCellCluster{ primaryVertexContext.getClusters()[iLayer][currentTracklet.firstClusterIndex] };
132  const Cluster& secondCellCluster{
133  primaryVertexContext.getClusters()[iLayer + 1][currentTracklet.secondClusterIndex]
134  };
135  const float firstCellClusterQuadraticRCoordinate{ firstCellCluster.rCoordinate * firstCellCluster.rCoordinate };
136  const float secondCellClusterQuadraticRCoordinate{ secondCellCluster.rCoordinate *
137  secondCellCluster.rCoordinate };
138  const float3 firstDeltaVector{ secondCellCluster.xCoordinate - firstCellCluster.xCoordinate,
139  secondCellCluster.yCoordinate - firstCellCluster.yCoordinate,
140  secondCellClusterQuadraticRCoordinate - firstCellClusterQuadraticRCoordinate };
141  const int nextLayerTrackletsNum{ static_cast<int>(primaryVertexContext.getTracklets()[iLayer + 1].size()) };
142 
143  for (int iNextLayerTracklet{ nextLayerFirstTrackletIndex };
144  iNextLayerTracklet < nextLayerTrackletsNum &&
145  primaryVertexContext.getTracklets()[iLayer + 1][iNextLayerTracklet].firstClusterIndex ==
146  nextLayerClusterIndex;
147  ++iNextLayerTracklet) {
148 
149  const Tracklet& nextTracklet{ primaryVertexContext.getTracklets()[iLayer + 1][iNextLayerTracklet] };
150  const float deltaTanLambda{ std::abs(currentTracklet.tanLambda - nextTracklet.tanLambda) };
151  const float deltaPhi{ std::abs(currentTracklet.phiCoordinate - nextTracklet.phiCoordinate) };
152 
153  if (deltaTanLambda < Constants::Thresholds::CellMaxDeltaTanLambdaThreshold &&
154  (deltaPhi < Constants::Thresholds::CellMaxDeltaPhiThreshold ||
155  std::abs(deltaPhi - Constants::Math::TwoPi) < Constants::Thresholds::CellMaxDeltaPhiThreshold)) {
156 
157  const float averageTanLambda{ 0.5f * (currentTracklet.tanLambda + nextTracklet.tanLambda) };
158  const float directionZIntersection{ -averageTanLambda * firstCellCluster.rCoordinate +
159  firstCellCluster.zCoordinate };
160  const float deltaZ{ std::abs(directionZIntersection - primaryVertex.z) };
161 
162  if (deltaZ < Constants::Thresholds::CellMaxDeltaZThreshold()[iLayer]) {
163 
164  const Cluster& thirdCellCluster{
165  primaryVertexContext.getClusters()[iLayer + 2][nextTracklet.secondClusterIndex]
166  };
167 
168  const float thirdCellClusterQuadraticRCoordinate{ thirdCellCluster.rCoordinate *
169  thirdCellCluster.rCoordinate };
170 
171  const float3 secondDeltaVector{ thirdCellCluster.xCoordinate - firstCellCluster.xCoordinate,
172  thirdCellCluster.yCoordinate - firstCellCluster.yCoordinate,
173  thirdCellClusterQuadraticRCoordinate -
174  firstCellClusterQuadraticRCoordinate };
175 
176  float3 cellPlaneNormalVector{ MathUtils::crossProduct(firstDeltaVector, secondDeltaVector) };
177 
178  const float vectorNorm{ std::sqrt(cellPlaneNormalVector.x * cellPlaneNormalVector.x +
179  cellPlaneNormalVector.y * cellPlaneNormalVector.y +
180  cellPlaneNormalVector.z * cellPlaneNormalVector.z) };
181 
182  if (vectorNorm < Constants::Math::FloatMinThreshold ||
183  std::abs(cellPlaneNormalVector.z) < Constants::Math::FloatMinThreshold) {
184 
185  continue;
186  }
187 
188  const float inverseVectorNorm{ 1.0f / vectorNorm };
189  const float3 normalizedPlaneVector{ cellPlaneNormalVector.x * inverseVectorNorm,
190  cellPlaneNormalVector.y * inverseVectorNorm,
191  cellPlaneNormalVector.z * inverseVectorNorm };
192  const float planeDistance{ -normalizedPlaneVector.x * (secondCellCluster.xCoordinate - primaryVertex.x) -
193  (normalizedPlaneVector.y * secondCellCluster.yCoordinate - primaryVertex.y) -
194  normalizedPlaneVector.z * secondCellClusterQuadraticRCoordinate };
195  const float normalizedPlaneVectorQuadraticZCoordinate{ normalizedPlaneVector.z * normalizedPlaneVector.z };
196  const float cellTrajectoryRadius{ std::sqrt(
197  (1.0f - normalizedPlaneVectorQuadraticZCoordinate - 4.0f * planeDistance * normalizedPlaneVector.z) /
198  (4.0f * normalizedPlaneVectorQuadraticZCoordinate)) };
199  const float2 circleCenter{ -0.5f * normalizedPlaneVector.x / normalizedPlaneVector.z,
200  -0.5f * normalizedPlaneVector.y / normalizedPlaneVector.z };
201  const float distanceOfClosestApproach{ std::abs(
202  cellTrajectoryRadius - std::sqrt(circleCenter.x * circleCenter.x + circleCenter.y * circleCenter.y)) };
203 
204  if (distanceOfClosestApproach >
205  Constants::Thresholds::CellMaxDistanceOfClosestApproachThreshold()[iLayer]) {
206 
207  continue;
208  }
209 
210  const float cellTrajectoryCurvature{ 1.0f / cellTrajectoryRadius };
211 
212  if (iLayer > 0 &&
213  primaryVertexContext.getCellsLookupTable()[iLayer - 1][iTracklet] == Constants::ITS::UnusedIndex) {
214 
215  primaryVertexContext.getCellsLookupTable()[iLayer - 1][iTracklet] =
216  primaryVertexContext.getCells()[iLayer].size();
217  }
218 
219  primaryVertexContext.getCells()[iLayer].emplace_back(
220  currentTracklet.firstClusterIndex, nextTracklet.firstClusterIndex, nextTracklet.secondClusterIndex,
221  iTracklet, iNextLayerTracklet, normalizedPlaneVector, cellTrajectoryCurvature);
222  }
223  }
224  }

397 template <bool IsGPU>
398 void Tracker<IsGPU>::findTracks(const Event& event)
399 {
400  mPrimaryVertexContext.getTracks().reserve(mPrimaryVertexContext.getRoads().size());
401  std::vector<TrackITS> tracks;
402  tracks.reserve(mPrimaryVertexContext.getRoads().size());
403  for (auto& road : mPrimaryVertexContext.getRoads()) {
404  std::array<int, 7> clusters{ Constants::ITS::UnusedIndex };
405  int lastCellLevel = Constants::ITS::UnusedIndex;
406  for (int iCell{ 0 }; iCell < Constants::ITS::CellsPerRoad; ++iCell) {
407  const int cellIndex = road[iCell];
408  if (cellIndex == Constants::ITS::UnusedIndex) {
409  continue;
410  } else {
411  clusters[iCell] = mPrimaryVertexContext.getCells()[iCell][cellIndex].getFirstClusterIndex();
412  clusters[iCell + 1] = mPrimaryVertexContext.getCells()[iCell][cellIndex].getSecondClusterIndex();
413  clusters[iCell + 2] = mPrimaryVertexContext.getCells()[iCell][cellIndex].getThirdClusterIndex();
414  lastCellLevel = iCell;
415  }
416  }
417  if (lastCellLevel == Constants::ITS::UnusedIndex)
418  continue;
419 
421  for (int iC{ 0 }; iC < clusters.size(); iC++) {
422  if (clusters[iC] != Constants::ITS::UnusedIndex) {
423  clusters[iC] = mPrimaryVertexContext.getClusters()[iC][clusters[iC]].clusterId;
424  }
425  }
427  const auto& cluster1_glo = event.getLayer(lastCellLevel + 2).getCluster(clusters[lastCellLevel + 2]);
428  const auto& cluster2_glo = event.getLayer(lastCellLevel + 1).getCluster(clusters[lastCellLevel + 1]);
429  const auto& cluster3_glo = event.getLayer(lastCellLevel).getCluster(clusters[lastCellLevel]);
430 
431  const auto& cluster3_tf = event.getLayer(lastCellLevel).getTrackingFrameInfo(clusters[lastCellLevel]);
432 
434  TrackITS temporaryTrack{ buildTrackSeed(cluster1_glo, cluster2_glo, cluster3_glo, cluster3_tf) };
435  for (size_t iC = 0; iC < clusters.size(); ++iC)
436  temporaryTrack.setExternalClusterIndex(iC, clusters[iC]);
437 
438  bool fitSuccess = fitTrack(event, temporaryTrack, Constants::ITS::LayersNumber - 4, -1, -1);
439  if (!fitSuccess)
440  continue;
441  temporaryTrack.resetCovariance();
442  fitSuccess = fitTrack(event, temporaryTrack, 0, Constants::ITS::LayersNumber, 1);
443  if (!fitSuccess)
444  continue;
445  temporaryTrack.getParamOut() = temporaryTrack;
446  temporaryTrack.resetCovariance();
447  fitSuccess = fitTrack(event, temporaryTrack, Constants::ITS::LayersNumber - 1, -1, -1);
448  if (!fitSuccess)
449  continue;
450  temporaryTrack.setROFrame(mROFrame);
451  tracks.emplace_back(temporaryTrack);
452  }
453 
454  std::sort(tracks.begin(), tracks.end(),
455  [](TrackITS& track1, TrackITS& track2) { return track1.isBetter(track2, 1.e6f); });
456 
457  for (auto& track : tracks) {
458  bool sharingCluster = false;
459  for (int iLayer{ 0 }; iLayer < Constants::ITS::LayersNumber; ++iLayer) {
460  if (track.getClusterIndex(iLayer) == Constants::ITS::UnusedIndex) {
461  continue;
462  }
463  sharingCluster |= mPrimaryVertexContext.isClusterUsed(iLayer, track.getClusterIndex(iLayer));
464  }
465  if (sharingCluster) {
466  continue;
467  }
468  for (int iLayer{ 0 }; iLayer < Constants::ITS::LayersNumber; ++iLayer) {
469  if (track.getClusterIndex(iLayer) == Constants::ITS::UnusedIndex) {
470  continue;
471  }
472  mPrimaryVertexContext.markUsedCluster(iLayer, track.getClusterIndex(iLayer));
473  }
474  mPrimaryVertexContext.getTracks().emplace_back(track);
475  }

478 template <bool IsGPU>
479 bool Tracker<IsGPU>::fitTrack(const Event& event, TrackITS& track, int start, int end, int step)
480 {
481  track.setChi2(0);
482  for (int iLayer{ start }; iLayer != end; iLayer += step) {
483  if (track.getClusterIndex(iLayer) == Constants::ITS::UnusedIndex) {
484  continue;
485  }
486  const TrackingFrameInfo& trackingHit = event.getLayer(iLayer).getTrackingFrameInfo(track.getClusterIndex(iLayer));
487 
488  if (!track.rotate(trackingHit.alphaTrackingFrame))
489  return false;
490 
491  if (!track.propagateTo(trackingHit.xTrackingFrame, getBz()))
492  return false;
493 
494  track.setChi2(track.getChi2() +
495  track.getPredictedChi2(trackingHit.positionTrackingFrame, trackingHit.covarianceTrackingFrame));
496  if (!track.TrackParCov::update(trackingHit.positionTrackingFrame, trackingHit.covarianceTrackingFrame))
497  return false;
498 
499  const float xx0 = (iLayer > 2) ? 0.008f : 0.003f; // Rough layer thickness
500  constexpr float radiationLength = 9.36f; // Radiation length of Si [cm]
501  constexpr float density = 2.33f; // Density of Si [g/cm^3]
502  if (!track.correctForMaterial(xx0, xx0 * radiationLength * density, true))
503  return false;
504  }
505  return true;
506 }

668 template <bool IsGPU>
669 track::TrackParCov Tracker<IsGPU>::buildTrackSeed(const Cluster& cluster1, const Cluster& cluster2,
670  const Cluster& cluster3, const TrackingFrameInfo& tf3)
671 {
672  const float ca = std::cos(tf3.alphaTrackingFrame), sa = std::sin(tf3.alphaTrackingFrame);
673  const float x1 = cluster1.xCoordinate * ca + cluster1.yCoordinate * sa;
674  const float y1 = -cluster1.xCoordinate * sa + cluster1.yCoordinate * ca;
675  const float z1 = cluster1.zCoordinate;
676  const float x2 = cluster2.xCoordinate * ca + cluster2.yCoordinate * sa;
677  const float y2 = -cluster2.xCoordinate * sa + cluster2.yCoordinate * ca;
678  const float z2 = cluster2.zCoordinate;
679  const float x3 = tf3.xTrackingFrame;
680  const float y3 = tf3.positionTrackingFrame[0];
681  const float z3 = tf3.positionTrackingFrame[1];
682 
683  const float crv = TrackingUtils::computeCurvature(x1, y1, x2, y2, x3, y3);
684  const float x0 = TrackingUtils::computeCurvatureCentreX(x1, y1, x2, y2, x3, y3);
685  const float tgl12 = TrackingUtils::computeTanDipAngle(x1, y1, x2, y2, z1, z2);
686  const float tgl23 = TrackingUtils::computeTanDipAngle(x2, y2, x3, y3, z2, z3);
687 
688  const float fy = 1. / (cluster2.rCoordinate - cluster3.rCoordinate);
689  const float& tz = fy;
690  const float cy = (TrackingUtils::computeCurvature(x1, y1, x2, y2 + Constants::ITS::Resolution, x3, y3) - crv) /
691  (Constants::ITS::Resolution * getBz() * constants::math::B2C) *
692  20.f; // FIXME: MS contribution to the cov[14] (*20 added)
693  constexpr float s2 = Constants::ITS::Resolution * Constants::ITS::Resolution;
694 
695  return track::TrackParCov(tf3.xTrackingFrame, tf3.alphaTrackingFrame,
696  { y3, z3, crv * (x3 - x0), 0.5f * (tgl12 + tgl23),
697  std::abs(getBz()) < constants::math::Almost0 ? constants::math::Almost0
698  : crv / (getBz() * constants::math::B2C) },
699  { s2, 0.f, s2, s2 * fy, 0.f, s2 * fy * fy, 0.f, s2 * tz, 0.f, s2 * tz * tz, s2 * cy, 0.f,
700  s2 * fy * cy, 0.f, s2 * cy * cy });
701 }
