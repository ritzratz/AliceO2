62 void svgSegmentation(const Segmentation& seg, SVGWriter& w, bool showdes, bool showdualsampas, bool showpads,
63  bool showpadchannels)
64 {
65  std::vector<Contour<double>> dualSampaContours = getDualSampaContours(seg);
66  std::vector<std::vector<Polygon<double>>> dualSampaPads = getPadPolygons(seg);
67  std::vector<std::vector<int>> dualSampaPadChannels = getPadChannels(seg);
68 
69  if (dualSampaPadChannels.size() != dualSampaPads.size()) {
70  throw std::runtime_error("gouze");
71  }
72 
73  auto deContour = getEnvelop(seg);
74  auto box = getBBox(seg);
75 
76  if (showpads) {
77  w.svgGroupStart("pads");
78  for (auto& dsp : dualSampaPads) {
79  for (auto& p : dsp) {
80  w.polygon(p);
81  }
82  }
83  w.svgGroupEnd();
84  }
85 
86  if (showpadchannels) {
87  w.svgGroupStart("padchannels");
88  for (auto i = 0; i < dualSampaPads.size(); ++i) {
89  auto& dsp = dualSampaPads[i];
90  auto& dspch = dualSampaPadChannels[i];
91  for (auto j = 0; j < dsp.size(); j++) {
92  auto bbox = getBBox(dsp[j]);
93  w.text(std::to_string(dspch[j]), bbox.xcenter(),
94  bbox.ymax() - 0.05 * bbox.height()); // SVG text y position is the bottom of the text
95  }
96  }
97  w.svgGroupEnd();
98  }
99 
100  if (showdualsampas) {
101  w.svgGroupStart("dualsampas");
102  for (auto& dsp : dualSampaContours) {
103  w.contour(dsp);
104  }
105  w.svgGroupEnd();
106  }
107 
108  if (showdes) {
109  w.svgGroupStart("detectionelements");
110  w.contour(deContour);
111  }
112 }
