97  bool dataGood = true;
98  for (int j = 0; j < header->payloadSize / sizeof(FakeCluster); ++j) {
99  float diff = std::abs(-inputDataTpc[j].x - inputDataTpcProcessed[j].x) +
100  std::abs(2 * inputDataTpc[j].y - inputDataTpcProcessed[j].y) +
101  std::abs(inputDataTpc[j].z * inputDataTpc[j].q - inputDataTpcProcessed[j].z) +
102  std::abs(inputDataTpc[j].q - inputDataTpcProcessed[j].q);
103  if (diff > 1) {
104  dataGood = false;
105  break;
106  }
107  }

109  LOG(INFO) << "qcTaskTPC - received data is " << (dataGood ? "correct" : "wrong");

237 void someProcessingStageAlgorithm(ProcessingContext& ctx)
238 {
239  const FakeCluster* inputDataTpc = reinterpret_cast<const FakeCluster*>(ctx.inputs().get("dataTPC").payload);
240  const FakeCluster* inputDataIts = reinterpret_cast<const FakeCluster*>(ctx.inputs().get("dataITS").payload);
241 
242  auto processedTpcClusters = ctx.allocator().make<FakeCluster>(OutputSpec{ "TPC", "CLUSTERS_P", 0 },
243  collectionChunkSize);
244  auto processedItsClusters = ctx.allocator().make<FakeCluster>(OutputSpec{ "ITS", "CLUSTERS_P", 0 },
245  collectionChunkSize);
246 
247  int i = 0;
248  for (auto& cluster : processedTpcClusters) {
249  assert(i < collectionChunkSize);
250  cluster.x = -inputDataTpc[i].x;
251  cluster.y = 2 * inputDataTpc[i].y;
252  cluster.z = inputDataTpc[i].z * inputDataTpc[i].q;
253  cluster.q = inputDataTpc[i].q;
254  i++;
255  }
256 
257  i = 0;
258  for (auto& cluster : processedItsClusters) {
259  assert(i < collectionChunkSize);
260  cluster.x = -inputDataIts[i].x;
261  cluster.y = 2 * inputDataIts[i].y;
262  cluster.z = inputDataIts[i].z * inputDataIts[i].q;
263  cluster.q = inputDataIts[i].q;
264  i++;
265  }
