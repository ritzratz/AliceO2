62 void DigitizerTask::Exec(Option_t* option)
63 {
64  FairRootManager* mgr = FairRootManager::Instance();
65 
66  if (mDigitsArray)
67  mDigitsArray->clear();
68  mDigitizer.setEventTime(mgr->GetEventTime());
69 
70  if (mMCTruthArray) {
71  mMCTruthArray->clear();
72  }
73 
74  o2::dataformats::MCTruthContainer<o2::tof::MCLabel> transientTruthContainer;
75  mDigitizer.setMCTruthContainer(&transientTruthContainer);
76 
77  // the type of digitization is steered by the DigiParams object of the Digitizer
78  LOG(DEBUG) << "Running digitization on new event " << mEventID << " from source " << mSourceID << FairLogger::endl;
79 
82  mDigitizer.setSrcID(mSourceID);
83  mDigitizer.setEventID(mEventID);
84 
85  LOG(INFO) << "Digitizing " << mHitsArray->size() << " hits \n";
86  mDigitizer.process(mHitsArray, mDigitsArray);
87 
88  // copying the transient labels to the output labels (stripping the tdc information)
89  if (mMCTruthArray) {
90  // copy from transientTruthContainer to mMCTruthAray
91  // a brute force solution for the moment; should be handled by a dedicated API
92  for (int index = 0; index < transientTruthContainer.getIndexedSize(); ++index) {
93  mMCTruthArray->addElements(index, transientTruthContainer.getLabels(index));
94  }
95  }
96 
97  mEventID++;
98 }
