39 void TrackITS::getImpactParams(Float_t x, Float_t y, Float_t z, Float_t bz, Float_t ip[2]) const
40 {
41  //------------------------------------------------------------------
42  // This function calculates the transverse and longitudinal impact parameters
43  // with respect to a point with global coordinates (x,y,0)
44  // in the magnetic field "bz" (kG)
45  //------------------------------------------------------------------
46  Float_t f1 = getSnp(), r1 = TMath::Sqrt((1. - f1) * (1. + f1));
47  Float_t xt = getX(), yt = getY();
48  Float_t sn = TMath::Sin(getAlpha()), cs = TMath::Cos(getAlpha());
49  Float_t a = x * cs + y * sn;
50  y = -x * sn + y * cs;
51  x = a;
52  xt -= x;
53  yt -= y;
54 
55  Float_t rp4 = getCurvature(bz);
56  if ((TMath::Abs(bz) < Almost0) || (TMath::Abs(rp4) < Almost0)) {
57  ip[0] = -(xt * f1 - yt * r1);
58  ip[1] = getZ() + (ip[0] * f1 - xt) / r1 * getTgl() - z;
59  return;
60  }
61 
62  sn = rp4 * xt - f1;
63  cs = rp4 * yt + r1;
64  a = 2 * (xt * f1 - yt * r1) - rp4 * (xt * xt + yt * yt);
65  Float_t rr = TMath::Sqrt(sn * sn + cs * cs);
66  ip[0] = -a / (1 + rr);
67  Float_t f2 = -sn / rr, r2 = TMath::Sqrt((1. - f2) * (1. + f2));
68  ip[1] = getZ() + getTgl() / rp4 * TMath::ASin(f2 * r1 - f1 * r2) - z;
69 }
