122 // find TGeoMedium instance given a detector prefix and a local (vmc) medium index
123 TGeoMedium* MaterialManager::getTGeoMedium(std::string const& modname, int localindex)
124 {
125  auto p = std::make_pair(modname, localindex);
126  auto iter = mTGeoMediumMap.find(p);
127  if (iter == mTGeoMediumMap.end()) {
128  LOG(WARNING) << "No medium registered for " << modname << " index " << localindex << "\n";
129  return nullptr;
130  }
131  return iter->second;
132 }
